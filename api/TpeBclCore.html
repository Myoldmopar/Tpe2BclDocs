def self.main(dir_to_process, key, product_name = 'DummyName', skip_unzipping = false)
  # process/validate command line argument 1: process directory
  raise ExCommandLineArgument, 'It appears no directory was passed as a command line argument ... aborting!' if dir_to_process.nil?
  raise ExCommandLineArgument, "Invalid directory passed in = '#{dir_to_process}' ... aborting!" unless File.directory?(dir_to_process)

  # process key
  if key.nil?
    raise ExCommandLineArgument, 'No product category passed in...aborting!'
  end

  # instantiate product data class
  product = SingleProductData.new(dir_to_process, key, product_name)

  # instantiate product processor class (also processes the product)
  processor = SingleProductProcessor.new(product, skip_unzipping)

  # create output stuff; first the performance data parameters
  performance_data = if product.data_from_csv.nil?
                       []
                     else
                       product.product_category.get_bcl_parameters(product)
                     end

  # close the osc and log before leaving
  $osm.close(key) unless product.skip_this_product

  # add in the openstudio version for kicks
  os_version = "#{$osm.osm_model.version.major}.#{$osm.osm_model.version.minor}.#{$osm.osm_model.version.patch}"
  $log.log("****OPENSTUDIO VERSION: #{os_version}")

  # the file paths this script generated
  payload_file_paths = []
  if $osm.osm_written
    payload_file_paths << { path: $osm.osm_file_path, file_type: 'OpenStudio', software_version: os_version }
  end

  # an informational comment -- most useful in the case of NO-GO!
  status_message = product.status_message

  # the flag for whether or not to push this
  go_no_go = !product.skip_this_product

  # create the actual list of items to return
  return_components = [performance_data, payload_file_paths, status_message, go_no_go]

  # for debugging
  if true
    for i in 0..3
      puts '**Return Item ' + i.to_s + ':'
      puts return_components[i]
    end
  end

  $log.log('Complete.  Thank you.', true)
  $log.close

  # then return them
  return_components
end
