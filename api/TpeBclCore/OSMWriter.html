def initialize(parent_dir)
  @parent_dir = parent_dir
  FileUtils.mkdir(@parent_dir) unless Dir.exist?(@parent_dir)
  @osm_file_path = File.join(@parent_dir, OSM_FILE)
  @osm_model = OpenStudio::Model::Model.new
  @osm_written = false
end

def close(key)
  # initialize obj to nothing, it will remain nil if we don't hit a valid entry, and assume we don't have an osc to write
  obj = nil
  @osm_written = true

  # switch on the key to get an obj instance to create a component out of; if we hit one that doesn't have osc support, obj will be nil
  case key
  when 'ndrnt' then  obj = $tu
  when 'tdrnt' then  obj = $acvrf
  when 'htpmpw' then obj = $hpwh
  when 'rftpnt' then obj = $rtu
  when 'nnslmp', 'slrplc', 'slmnrs', 'nnslmn' then obj = $lights
  when 'gsfrdn' then obj = $unit_heater
  when 'nvrtr' then  obj = $inverter
  when 'htwtrb' then obj = $boiler
  when 'phtvlt' then obj = $pv
  when 'rtdynm' then obj = $pump
  when 'stmblr' then obj = $boiler
  when 'rfrgrt' then obj = $rcase
  when 'lwvltg' then # no OpenStudio object for Transformer yet; see https://github.com/NREL/OpenStudio/blob/6c68e04c9585ee5431e2695a4c1135bdae215271/openstudiocore/src/model/ElectricLoadCenterDistribution.hpp#L46;
  else; raise ExProductCategoryNotImplemented, 'Product did not have a close() entry for OpenStudio file output; either make an explicit block in the switch to ignore it, or add OS support'
  end

  # and then if we ended up with a valid obj, we must have an osc to write, so do it!
  if obj
    @osm_written = true
    obj.createComponent.save(@osm_file_path)
  end
end

def write_one_indep_var_table(xy_data, unit_types, curve_type = 'Cubic')
  # make sure there is *something* there first
  if xy_data.empty?
    $log.log('Attempted to write one indep var table with empty array, returning quietly')
    return
  end

  # check the values first and also get min/max values
  minx = 9_999_999_999
  miny = 9_999_999_999
  maxx = -9_999_999_999
  maxy = -9_999_999_999
  xy_data.each do |xy|
    if xy.nil?
      return
    elsif xy[0].nil?
      return
    elsif xy[1].nil?
      return
    end
    minx = [minx, xy[0]].min
    miny = [miny, xy[1]].min
    maxx = [maxx, xy[0]].max
    maxy = [maxy, xy[1]].max
  end

  this_table = OpenStudio::Model::TableMultiVariableLookup.new($osm.osm_model, 1)
  this_table.setCurveType(curve_type.to_s)
  this_table.setInterpolationMethod('LinearInterpolationOfTable')
  this_table.setMinimumValueofX1(minx)
  this_table.setMaximumValueofX1(maxx)
  this_table.setMinimumTableOutput(miny)
  this_table.setMaximumTableOutput(maxy)
  this_table.setInputUnitTypeforX1(unit_types[0])
  this_table.setOutputUnitType(unit_types[1])
  this_table.setNormalizationReference(1.0)
  xy_data.each do |xy|
    this_table.addPoint(xy[0], xy[1])
  end

  this_table
end

def write_quadratic_curve(coefficients, x_bounds, y_bounds, unit_types)
  this_curve = OpenStudio::Model::CurveQuadratic.new($osm.osm_model)
  this_curve.setCoefficient1Constant(coefficients[0])
  this_curve.setCoefficient2x(coefficients[1])
  this_curve.setCoefficient3xPOW2(coefficients[2])
  this_curve.setMinimumValueofx(x_bounds[0])
  this_curve.setMaximumValueofx(x_bounds[1])
  this_curve.setMinimumCurveOutput(y_bounds[0])
  this_curve.setMaximumCurveOutput(y_bounds[1])
  this_curve.setInputUnitTypeforX(unit_types[0])
  this_curve.setOutputUnitType(unit_types[1])

  this_curve
end
