def initialize(parent_dir)
    @parent_dir = parent_dir
    unless Dir.exist?(@parent_dir)
        FileUtils.mkdir(@parent_dir)
    end
    @osm_file_path = File.join(@parent_dir, OSM_FILE)
    @osm_model = OpenStudio::Model::Model.new
    @osm_written = false
end

def close(key)
    if key == "ndrnt"
        @osm_model.getZoneHVACTerminalUnitVariableRefrigerantFlows()[0].createComponent.save(@osm_file_path)
        @osm_written = true
    elsif key == "tdrnt"
        @osm_model.getAirConditionerVariableRefrigerantFlows()[0].createComponent.save(@osm_file_path)
        @osm_written = true
    elsif key == "htpmpw"
        @osm_model.getWaterHeaterHeatPumps()[0].createComponent.save(@osm_file_path)
        @osm_written = true
    elsif key == "rftpnt"
        @osm_model.getAirLoopHVACUnitarySystems()[0].createComponent.save(@osm_file_path)
        @osm_written = true
    elsif ["nnslmp","slrplc","slmnrs","nnslmn"].include? key
        @osm_model.getLightss()[0].createComponent.save(@osm_file_path)
        @osm_written = true
    else
        # don't do anything until they have OSM capability
    end
end

def write_one_indep_var_table(xy_data, unit_types, curve_type = "Cubic")

    # make sure there is *something* there first
    if xy_data.length < 1
        $log.log("Attempted to write one indep var table with empty array, returning quietly")
        return
    end

    # check the values first and also get min/max values
    minx, miny = 9999999999, 9999999999
    maxx, maxy = -9999999999, -9999999999
    xy_data.each do |xy|
        if xy.nil?
            return
        elsif xy[0].nil?
            return
        elsif xy[1].nil?
            return
        end
        minx = [minx, xy[0]].min
        miny = [miny, xy[1]].min
        maxx = [maxx, xy[0]].max
        maxy = [maxy, xy[1]].max
    end

    this_table = OpenStudio::Model::TableMultiVariableLookup.new($osm.osm_model, 1)
    this_table.setCurveType("#{curve_type}")
    this_table.setInterpolationMethod("LinearInterpolationOfTable")
    this_table.setMinimumValueofX1(minx)
    this_table.setMaximumValueofX1(maxx)
    this_table.setMinimumTableOutput(miny)
    this_table.setMaximumTableOutput(maxy)
    this_table.setInputUnitTypeforX1(unit_types[0])
    this_table.setOutputUnitType(unit_types[1])
    this_table.setNormalizationReference(1.0)
    xy_data.each do |xy|
        this_table.addPoint(xy[0], xy[1])
    end

    return this_table
end

def write_quadratic_curve(coefficients, x_bounds, y_bounds, unit_types)

    this_curve = OpenStudio::Model::CurveQuadratic.new($osm.osm_model)
    this_curve.setCoefficient1Constant(coefficients[0])
    this_curve.setCoefficient2x(coefficients[1])
    this_curve.setCoefficient3xPOW2(coefficients[2])
    this_curve.setMinimumValueofx(x_bounds[0])
    this_curve.setMaximumValueofx(x_bounds[1])
    this_curve.setMinimumCurveOutput(y_bounds[0])
    this_curve.setMaximumCurveOutput(y_bounds[1])
    this_curve.setInputUnitTypeforX(unit_types[0])
    this_curve.setOutputUnitType(unit_types[1])

    return this_curve

end

    # make sure there is *something* there first
    if xyz_data.length < 1
        $log.log("Attempted to write two indep var table with empty array, returning quietly")
        return
    end

    # check the values first and also get min/max values
    minx, miny, minz = 9999999999, 9999999999, 9999999999
    maxx, maxy, maxz = -9999999999, -9999999999, -9999999999
    xyz_data.each do |xyz|
        if xyz.nil?
            return
        elsif xyz[0].nil?
            return
        elsif xyz[1].nil?
            return
        elsif xyz[2].nil?
            return
        end
        minx = [minx, xyz[0]].min
        miny = [miny, xyz[1]].min

        maxx = [maxx, xyz[0]].max
        maxy = [maxy, xyz[1]].max

    end

    this_table.setCurveType("BiQuadratic")
    this_table.setInterpolationMethod("LinearInterpolationOfTable")

    this_table.setInputUnitTypeforX2(unit_types[1])
    this_table.setOutputUnitType(unit_types[2])
    this_table.setNormalizationReference(1.0)
