def self.idf(message)
    
    if @@output_dir.nil?
        self.log("Diagnostic error in Out.idf", true) 
        self.log("Attempted to write to idf file, but output directory has not been set.", true)
        self.log("Output directory should be set early in the script, by the ProductProcessor", true)
        self.log("Likely an implementation error, not an input error; aborting...", true)
        exit ExitCodes::DIAG_INVALIDOUTPUTFOLDER
    end
    
    unless Dir.exist?(@@output_dir)
        FileUtils.mkdir(@@output_dir) 
    end
    
    if @@idf_file.nil?
        
        @@idf_file_name = "#{@@output_dir}/#{IDF_FILE}"
        @@idf_file = File.open(@@idf_file_name, 'w')
        @@idf_file.puts("! TPE -> BCL IDF OBJECT OUTPUT FILE")
        @@idf_file.puts("! Scripts written by Edwin Lee, Katherine Fleming, and Daniel Studer")
        
    end
    
    @@idf_file.puts(message)
    
end

def self.log(message, also_to_console=false)
    
    if @@output_dir.nil?
        puts "Diagnostic error in Out.log"
        puts "Attempted to write to log file, but output directory has not been set."
        puts "Output directory should be set early in the script, by the ProductProcessor"
        puts "Thus, log function can't be called before product processing has begun"
        puts "Likely an implementation error, not an input error; aborting..."
        puts "Call stack: "
        puts caller
        exit ExitCodes::DIAG_INVALIDOUTPUTFOLDER
    end
    
    time = Time.new
    if @@log_file.nil?
        
        unless Dir.exist?(@@output_dir)
            FileUtils.mkdir(@@output_dir) 
        end
                    
        if @@time_stamp.nil?
            @@time_stamp = time.strftime("%Y%m%d%H%M%S") 
        end
        
        @@log_file_name = "#{@@output_dir}/#{@@time_stamp}.log"
        @@log_file = File.open(@@log_file_name, 'w')
        @@log_file.puts("TPE -> BCL PRODUCT LOG FILE")
        @@log_file.puts("Scripts written by Edwin Lee, Katherine Fleming, and Daniel Studer")
        
    end
    
    @@log_file.puts(self.prefix + message)
    if also_to_console
        puts message
    end

end

def self.write_curve_quadlinear(vector_x, curve_name)
    
    self.idf("
Curve:QuadLinear,
   !\memo curve = C1 + C2*w + C3*x + C4*y + C5*z
    #{curve_name},  !- Name
    #{vector_x[0,0]},  !- Coefficient1 Constant
    #{vector_x[1,0]},  !- Coefficient2 w
    #{vector_x[2,0]},  !- Coefficient3 x
    0,        !- Coefficient4 y
    0,        !- Coefficient5 z
    0,        !- Minimum Value of w
    100,      !- Maximum Value of w
    0,        !- Minimum Value of x
    1.0,      !- Maximum Value of x
    0,        !- Minimum Value of y
    1,        !- Maximum Value of y
    0,        !- Minimum Value of z
    1,        !- Maximum Value of z")

end

def self.write_one_indep_var_table(xy_data, table_name, unit_types)

    # write the bulk of the idf object
    self.idf("
Table:OneIndependentVariables,
    #{table_name},   !- Name
    Cubic,     !- Curve Type
    LinearInterpolationOfTable, !- Interpolation Type
    0.0,  !- Minimum Value of x,
    1000, !- Maximum Value of x,
    0.0, !- Minimum Table Output
    1000, !- Maximum Table Output
    #{unit_types[0]}, !- Input Unit Type for x
    #{unit_types[1]}, !- Output Unit Type
    1.0, !- Normalization Point")
    
    # write all the values except the last one
    xy_data[0..-2].each do |xy|
        self.idf("    #{xy[0]}, #{xy[1]},")
    end
    
    # write the final value terminating it with a semicolon
    self.idf("    #{xy_data[-1][0]}, #{xy_data[-1][1]};")

end
