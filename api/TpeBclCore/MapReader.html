def initialize(file_full_path, product_category, map_source_type, csv_data, map_file_name)
  $log.log("Processing map at path: #{file_full_path}")

  # open the spreadsheet
  if file_full_path.end_with?('.xls')
    @spreadsheet_instance = Roo::Excel.new(file_full_path)
  elsif file_full_path.end_with?('.xlsx')
    @spreadsheet_instance = Roo::Excelx.new(file_full_path)
  end

  # instantiate a map
  @this_map = product_category::Map.new(@spreadsheet_instance, csv_data, map_file_name)
  @map_source_type = map_source_type
  @insufficient_data_points = @this_map.insufficient_data_points
  @this_map_data_set = @this_map.this_data_set
end
