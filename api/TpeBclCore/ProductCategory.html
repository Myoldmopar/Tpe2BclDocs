def self.get_category_index_from_string(string_category)
    IMPLEMENTED_CATEGORIES.each { |x|
        if string_category.downcase == x.string_key
            return x.index_number
        end            
    }
    Out.log("Invalid product category string found in xml ...", true)
    Out.log("Entered string = '#{string_category}' ...", true)
    Out.log("Allowable entries include (case-INsensitive): ...", true)
    ProductCategory.list_allowed_categories
    exit ExitCodes::ERR_INVALIDPRODCATEGORY
end

def self.get_parameter_listings_for_category(product_category)
    IMPLEMENTED_CATEGORIES.each { |x|
        if product_category == x.index_number
            return x.parameter_list
        end
    }
    Out.log("Implementation Error: Invalid product category index in ProductCategory.get_parameter_listings_for_category ...", true)
    Out.log("Argument index = '#{product_category.to_s}' ...", true)
    Out.log("Allowable indices include: ...", true)
    ProductCategory.list_allowed_keys
    exit ExitCodes::DIAG_INVALIDPRODCATEGORY
end
