def self.linear_regression(x, y)
  x_avg = x.reduce(0) { |sum, x| x + sum }.to_f / x.length
  y_avg = y.reduce(0) { |sum, y| y + sum }.to_f / y.length
  numerator = (0...x.length).reduce(0) { |sum, i| sum + ((x[i] - x_avg) * (y[i] - y_avg)) }
  denominator = x.reduce(0) { |sum, x| sum + ((x - x_avg)**2) }
  slope = numerator / denominator
  y_intercept = y_avg - (slope * x_avg)
  [y_intercept, slope]
end

def self.strip_degree_units(value)
  if value.nil?
    return nil
  elsif parameter_is_numeric?(value)
    return Float(value)
  elsif value.include?("°")
    value_corrected = value.tr("–", '-')
    # puts "Interpreting degree value"
    i = value_corrected.index("°")
    # puts "degree index = #{i}"
    potential_number = value_corrected[0..i - 1]
    # puts "potential number value = #{potential_number}"
    if parameter_is_numeric?(potential_number)
      newval = Float(potential_number)
      $log.log("String containing degree symbol found and successfully processed.  Original string = #{value}; processed numeric = #{newval}")
      return newval
    else
      $log.log("String containing degree symbol found, but couldn't be stripped properly; full string = #{value}")
      return nil
    end
  else
    return nil
  end
end
