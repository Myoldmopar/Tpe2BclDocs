def initialize(map_spreadsheet_object, csv_data, map_file_name)
  # read in the meta data
  owner = map_spreadsheet_object.cell(3, 4)
  brand        = map_spreadsheet_object.cell(4, 4)
  product_line = map_spreadsheet_object.cell(5, 4)
  model        = map_spreadsheet_object.cell(6, 4)

  # instantiate the base class with common data
  super(owner, brand, product_line, model)
  @type_clg_or_htg_or_defrost = nil

  # next we need to read in HPWH specific data
  # first instantiate the main data point array
  @this_data_set = []

  # get the rated COP and capacity first to scale the values in the map
  rated_cop = csv_data.get_parameter_average_value_by_descriptor(PARAM_RATEDCOP)
  rated_cap = csv_data.get_parameter_average_value_by_descriptor(PARAM_RATEDCAPACITY)

  # set reading bounds and other parameters, and start reading data
  first_row_of_data = 11
  last_row_of_data  = 40
  column_capacity   = 4
  column_power_in   = 5
  for row in first_row_of_data..last_row_of_data

    # create array for independent and dependent variables
    independent_vars = []
    dependent_vars = []

    # deduce air entering temp algorithmically...user CANT change the template
    case row
    when 11, 17, 23, 29, 35 then independent_vars << -9.4
    when 12, 18, 24, 30, 36 then independent_vars <<  2.8
    when 13, 19, 25, 31, 37 then independent_vars <<  6.4
    when 14, 20, 26, 32, 38 then independent_vars <<  13.6
    when 15, 21, 27, 33, 39 then independent_vars <<  21.8
    when 16, 22, 28, 34, 40 then independent_vars <<  23.9
    end

    # deduce condensing temp algorithmically...user CANT change the template
    case row
    when 11..16 then independent_vars << 12.8
    when 17..22 then independent_vars << 26.7
    when 23..28 then independent_vars << 37.8
    when 29..34 then independent_vars << 48.9
    when 35..40 then independent_vars << 60.0
    end

    # read the dependent variables
    capacity = map_spreadsheet_object.cell(row, column_capacity)
    power_in = map_spreadsheet_object.cell(row, column_power_in)

    valid_point = false
    for item in [capacity, power_in]
      if item.nil?
      # blank, leave it alone for now
      elsif !Numerics.parameter_is_numeric?(item)
        # non-numerics, skip it
        $log.log("Found non-numerics in map data: [#{capacity}, #{power_in}]")
        valid_point = false
        break
      else
        # good to go
        valid_point = true
      end
    end
    next unless valid_point

    # calculate the COP for this state
    this_cop = nil
    this_cop = capacity / power_in if !capacity.nil? && !power_in.nil?

    # scale from rated conditions
    dependent_vars << if capacity.nil?
                        nil
                      else
                        capacity / rated_cap
                      end
    dependent_vars << if this_cop.nil?
                        nil
                      else
                        this_cop / rated_cop
                      end

    # create a new data point
    this_data_point = HPWHMapDataPoint.new(independent_vars, dependent_vars)

    # add it to the main dataset hash
    @this_data_set.push(this_data_point)

  end

  spew('') # print base map data
  HPWHMapDataPoint.spew # print class-level header line
  @this_data_set.each(&:spew) # print all data points data

  # throw an error if the number of data points is not sufficient
  map_template_max_data_points = 30
  return unless @this_data_set.length.to_f / map_template_max_data_points.to_f < 0.25
  $log.log("Insufficient data points for found this map file: #{map_file_name}")
  @insufficient_data_points = true

end
