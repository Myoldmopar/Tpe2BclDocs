def self.get_bcl_parameters(product_data)
  parameters = {}
  params_to_include = [PARAM_RATEDPOWER,
                       PARAM_PHASE,
                       PARAM_CONDUCTORMATERIAL,
                       PARAM_FULLLOADTEMPRISE,
                       PARAM_EDDYCURRENTLOADLOSSFRAC,
                       PARAM_NOLOADLOSS,
                       PARAM_RATEDLOADLOSS,
                       PARAM_NAMEPLATEEFFICIENCY,
                       PARAM_MAXEFFICIENCYPOINT]
  params_to_include.each do |x|
    val = product_data.data_from_csv.get_parameter_average_value_by_descriptor(x)
    if val.nil?
      $log.log("Key BCL parameter, named '" + x + "', was not found in the input.  Skipping...")
    else
      parameters[x] = val
    end
  end
  parameters
end
