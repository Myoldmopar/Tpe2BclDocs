def self.get_bcl_parameters(product_data)
    parameters = {}
    params_to_include = [PARAM_INPUTPOWER]
    params_to_include.each do |x|
        parameters[x] = product_data.data_from_csv.get_parameter_average_value_by_descriptor(x)
    end
    return parameters
end

def self.suitable_for_bcl?(product_data)
    if product_data.data_from_csv.nil?
        return false
    end
    if product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_INPUTPOWER).nil?
        return false
    else
        return true
    end
end

def self.write_idf_stuff(product_data)

    lighting_watts = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_INPUTPOWER)

    lights_def = OpenStudio::Model::LightsDefinition.new($osm.osm_model)
    lights_def.setLightingLevel(lighting_watts)
    lights = OpenStudio::Model::Lights.new(lights_def)

    $idf.idf("
  Lights,
    ,                   !- Name
    ,                   !- Zone or ZoneList Name
    ,                   !- Schedule Name
    LightingLevel,      !- Design Level Calculation Method
    #{lighting_watts},  !- Lighting Level {W}
    ,                   !- Watts per Zone Floor Area {W/m2}
    ,                   !- Watts per Person {W/person}
    ,                   !- Return Air Fraction
    ,                   !- Fraction Radiant
    ,                   !- Fraction Visible
    ,                   !- Fraction Replaceable
    ;                   !- End-Use Subcategory
    ")

end
