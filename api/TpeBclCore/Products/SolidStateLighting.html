def self.get_bcl_parameters(product_data)
  parameters = {}
  params_to_include = [PARAM_INPUTPOWER]
  params_to_include.each do |x|
    parameters[x] = product_data.data_from_csv.get_parameter_average_value_by_descriptor(x)
  end
  parameters
end

def self.write_component(product_data)
  lighting_watts = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_INPUTPOWER)

  lights_def = OpenStudio::Model::LightsDefinition.new($osm.osm_model)
  lights_def.setLightingLevel(lighting_watts)
  $lights = OpenStudio::Model::Lights.new(lights_def)
end
