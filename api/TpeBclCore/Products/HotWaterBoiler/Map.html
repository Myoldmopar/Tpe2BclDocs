def initialize(map_spreadsheet_object, _csv_data, map_file_name)
  # read in the meta data
  owner        = map_spreadsheet_object.cell(3, 3)
  brand        = map_spreadsheet_object.cell(4, 3)
  product_line = map_spreadsheet_object.cell(5, 3)
  model        = map_spreadsheet_object.cell(6, 3)

  # instantiate the base class with common data
  super(owner, brand, product_line, model)

  # next read in hot-water boiler specific data
  # first instantiate the main data point array
  @this_data_set = []

  # get detailed data
  condensing_operation   =  map_spreadsheet_object.cell(9, 3)
  minimum_flow_rate      = (map_spreadsheet_object.cell(10, 3) / 15_852) # convert from gpm to m3/s
  maximum_flow_rate      = (map_spreadsheet_object.cell(11, 3) / 15_852)
  minimum_feedwater_temp = (map_spreadsheet_object.cell(12, 3) - 32) * (5 / 9) # convert to celsius
  input_power_control    =  map_spreadsheet_object.cell(13, 3)
  if input_power_control == 'Modulating'
    num_discrete_stages  = 4
  else
    num_discrete_stages  = input_power_control == 'Single Stage' ? 1 : map_spreadsheet_object.cell(14, 3)
    num_discrete_stages  = Numerics.parameter_is_numeric?(num_discrete_stages) ? num_discrete_stages : num_discrete_stages.delete('^0-9.').to_f # check if >=4 and if so change to 4
  end

  # set reading bounds and other parameters, and start reading data
  first_row_of_data = 18
  num_rows = condensing_operation == 'Yes' ? num_discrete_stages * 12 : num_discrete_stages * 9
  last_row_of_data = first_row_of_data + num_rows - 1
  column_flow_rate           = 3
  column_entering_water_temp = 4
  column_input_power         = 6
  column_leaving_water_temp  = 7

  for row in first_row_of_data..last_row_of_data

    # create array for independent and dependent variables
    independent_vars = []
    dependent_vars   = []

    # read in independent variables
    flow_rate = map_spreadsheet_object.cell(row, column_flow_rate)
    independent_vars << (flow_rate / 15_852) # convert from gpm to m3/s
    entering_water_temp = map_spreadsheet_object.cell(row, column_entering_water_temp)
    independent_vars << (entering_water_temp - 32) * 5 / 9 # convert to celsius

    # read in dependent variables
    input_power = map_spreadsheet_object.cell(row, column_input_power)
    dependent_vars << if input_power.nil?
      input_power
    else
      (input_power * 0.2930710701722) # convert MBh to kW
                      end
    leaving_water_temp = map_spreadsheet_object.cell(row, column_leaving_water_temp)
    dependent_vars << if leaving_water_temp.nil?
      input_power
    else
      (leaving_water_temp - 32) * 5 / 9 # convert to celsius
                      end

    valid_point = true
    for item in dependent_vars
      if item.nil?
        # blank, leave it alone for now
      elsif !Numerics.parameter_is_numeric?(item)
        # non-numerics, skip it
        $log.log("Found non-numerics in map data: #{dependent_vars}")
        valid_point = false
        # good to go
      end
    end
    next unless valid_point

    # create a new data point
    this_data_point = HotWaterBoilerDataPoint.new(independent_vars, dependent_vars)

    # add it to the main dataset hash
    @this_data_set.push(this_data_point)

  end

  spew('') # print base map data
  HotWaterBoilerDataPoint.spew # print class-level header line
  @this_data_set.each(&:spew) # prints all data points data

  # return if we have sufficient data points, otherwise report the problem
  map_template_max_data_points = 48
  return unless @this_data_set.length.to_f / map_template_max_data_points.to_f < 0.18
  $log.log("Insufficient data points found for this map file: #{map_file_name}")
  @insufficient_data_points = true

end
