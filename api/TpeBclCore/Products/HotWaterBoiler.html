def self.get_bcl_parameters(product_data)
  parameters = {}
  params_to_include = [PARAM_INPUTPOWER,
                       PARAM_THERMALEFFICIENCY,
                       PARAM_FUELTYPE,
                       PARAM_MAXFLOWRATE]
  params_to_include.each do |x|
    val = product_data.data_from_csv.get_parameter_average_value_by_descriptor(x)
    if val.nil?
      $log.log("Key BCL parameter, named '" + x + "', was not found in the input.  Skipping...")
    else
      parameters[x] = val
    end
  end
end

def self.write_base_component_parameters(product_data)
  fuel_type = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_FUELTYPE)
  input_power = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_INPUTPOWER)
  thermal_efficiency = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_THERMALEFFICIENCY)
  max_flow_rate = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_MAXFLOWRATE)
  nominal_capacity = nil # initialize
  if input_power && thermal_efficiency
    nominal_capacity = input_power * thermal_efficiency * 0.01
  end

  $boiler = OpenStudio::Model::BoilerHotWater.new($osm.osm_model)
  $boiler.setName(product_data.product_name)
  $boiler.setFuelType(fuel_type) unless fuel_type.nil?
  $boiler.setNominalCapacity(nominal_capacity) unless nominal_capacity.nil?
  $boiler.setEfficiencyCurveTemperatureEvaluationVariable('EnteringBoiler')
  $boiler.setNominalThermalEfficiency(thermal_efficiency) unless thermal_efficiency.nil?
  $boiler.setDesignWaterFlowRate(max_flow_rate) unless max_flow_rate.nil?
end

def self.write_component(product_data)
  # first write base component
  product_data.product_category.write_base_component_parameters(product_data)

  # merge multiple maps
  merged_data_set = TpeBclCore::MapMethods.merge_multiple_maps(product_data.data_from_maps, '', TpeBclCore::Products::HotWaterBoiler::Map::HotWaterBoilerDataPoint)

  # write map components
  product_data.product_category.write_map_component_parameters(product_data, merged_data_set)
end

def self.write_map_component_parameters(product_data, merged_data_set)
  units_array = [OSMWriter::UNIT_TYPE_DIMENSIONLESS, OSMWriter::UNIT_TYPE_TEMPERATURE, OSMWriter::UNIT_TYPE_DIMENSIONLESS]

  input_power = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_INPUTPOWER)
  thermal_efficiency = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_THERMALEFFICIENCY) * 0.01 # convert to decimal
  qnom = input_power * thermal_efficiency
  values = []
  merged_data_set.each do |datum|
    q = (datum.flow_rate * 968) * 4.203 * (datum.leaving_water_temp - datum.entering_water_temp) # q = (m/t)*Cp*dT
    plr = q / qnom
    efficiency = q / datum.input_power
    normalized_efficiency = efficiency / thermal_efficiency
    values.push([plr, datum.entering_water_temp, normalized_efficiency])
  end
  table_curve = $osm.write_two_indep_var_table(values, units_array)
  $boiler.setNormalizedBoilerEfficiencyCurve(table_curve)
end
