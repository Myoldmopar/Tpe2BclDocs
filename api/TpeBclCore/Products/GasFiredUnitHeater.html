def self.get_bcl_parameters(product_data)
    parameters = {}
    params_to_include = [PARAM_RATEDHEATCAP,
                         PARAM_GASEFFICIENCY,
                         PARAM_RATEDSUPPLYAIRFLOW,
                         PARAM_FUELTYPE,
                         PARAM_FANCONTROL,
                         PARAM_PRESSURERISE,
                         PARAM_MOTOREFFICIENCY]
    params_to_include.each do |x|
        val = product_data.data_from_csv.get_parameter_average_value_by_descriptor(x)
        if val.nil?
            $log.log("Key BCL parameter, named '" + x + "', was not found in the input.  Skipping...")
        else
            parameters[x] = val
        end
    end
    return parameters
end

def self.suitable_for_bcl?(product_data)
  required_parameters = [PARAM_RATEDHEATCAP,
                         PARAM_GASEFFICIENCY,
                         PARAM_RATEDSUPPLYAIRFLOW,
                         PARAM_FANCONTROL,
                         PARAM_PRESSURERISE,
                         PARAM_MOTOREFFICIENCY]
    list_of_missing_parameters = []
    required_parameters.each do |param|
        if product_data.data_from_csv.get_parameter_average_value_by_descriptor(param).nil?
            list_of_missing_parameters.push(param)
        end
    end
    if list_of_missing_parameters.length.zero?
        return true
    else
        product_data.status_message = $log.issue_missing_parameter_error(list_of_missing_parameters)
        return false
    end
end

def self.write_idf_stuff(product_data)

  # get parameters first
  rated_flow_rate = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_RATEDSUPPLYAIRFLOW)
  if rated_flow_rate.nil?
    rated_flow_rate = false
    rated_flow_rate_string = ""
  else
    rated_flow_rate_string = rated_flow_rate.to_s
  end

  fan_control = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_FANCONTROL)
  if fan_control.nil?
    fan_control = false
    fan_control_string = ""
  else
    fan_control_string = fan_control.to_s
  end

  pressure_rise = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_PRESSURERISE)
  if pressure_rise.nil?
    pressure_rise = false
    pressure_rise_string = ""
  else
    pressure_rise_string = pressure_rise.to_s
  end

  motor_efficiency = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_MOTOREFFICIENCY)
  if motor_efficiency.nil?
    motor_efficiency = false
    motor_efficiency_string = ""
  else
    motor_efficiency_string = (motor_efficiency * 0.01).to_s
  end

  gas_efficiency = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_GASEFFICIENCY)
  if gas_efficiency.nil?
    gas_efficiency = false
    gas_efficiency_string = ""
  else
    gas_efficiency_string = (gas_efficiency * 0.01).to_s
  end

  rated_heat_cap = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_RATEDHEATCAP)
  if rated_heat_cap.nil?
    rated_heat_cap = false
    rated_heat_cap_string = ""
  else
    rated_heat_cap_string = rated_heat_cap.to_s
  end

    $idf.idf("
ZoneHVAC:UnitHeater,
  #{product_data.product_name}, !- Name
  ,                             !- Availability Schedule Name
  ,                             !- Air Inlet Node Name
  ,                             !- Air Outlet Node Name
  Fan:ConstantVolume,           !- Supply Air Fan Object Type
  Zone1UnitHeatFan,             !- Fan Name
  #{rated_flow_rate_string},    !- Maximum Supply Air Flow Rate (m3/s)
  #{fan_control_string},        !- Fan Control Type
  Coil:Heating:Gas,             !- Heating Coil Object Type
  Zone1UnitHeatHeatingCoil,     !- Heating Coil Name
  ,                             !- Maximum Hot Water or Steam Flow Rate (m3/s)
  ,                             !- Minimum Hot Water or Steam Flow Rate (m3/s)
  ;                             !- Heating Convergence Tolerance

Fan:ConstantVolume,
  Zone1UnitHeatFan,             !- Name
  ,                             !- Availability Schedule Name
  ,                             !- Fan Total Efficiency
  #{pressure_rise_string},      !- Pressure Rise (Pa)
  #{rated_flow_rate_string},    !- Maximum Flow Rate (m3/s)
  #{motor_efficiency_string},   !- Motor Efficiency
  ,                             !- Motor in Airstream Fraction
  ,                             !- Fan Inlet Node Name
  ;                             !- Fan Outlet Node Name

Coil:Heating:Gas,
  Zone1UnitHeatHeatingCoil,     !- Name
  ,                             !- Availability Schedule Name
  #{gas_efficiency_string},     !- Gas Burner Efficiency
  #{rated_heat_cap_string},     !- Nominal Capacity (W)
  ,                             !- Air Inlet Node Name
  ;                             !- Air Outlet Node Name
          ")
