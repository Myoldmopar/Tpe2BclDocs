def self.get_bcl_parameters(product_data)
  parameters = {}
  params_to_include = [PARAM_RATEDHEATCAP,
                       PARAM_GASEFFICIENCY,
                       PARAM_RATEDSUPPLYAIRFLOW,
                       PARAM_FUELTYPE,
                       PARAM_FANCONTROL,
                       PARAM_PRESSURERISE,
                       PARAM_MOTOREFFICIENCY]
  params_to_include.each do |x|
    val = product_data.data_from_csv.get_parameter_average_value_by_descriptor(x)
    if val.nil?
      $log.log("Key BCL parameter, named '" + x + "', was not found in the input.  Skipping...")
    else
      parameters[x] = val
    end
  end
  parameters
end

def self.write_component(product_data)
  # the rated flow rate, gas burn efficiency, and heating capacity really should be there, they are pretty core pieces of information
  rated_flow_rate = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_RATEDSUPPLYAIRFLOW)
  gas_efficiency = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_GASEFFICIENCY)
  rated_heat_cap = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_RATEDHEATCAP)

  # these can be nil, since nil is falsy and we'll check each as we write them
  fan_control = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_FANCONTROL)
  pressure_rise = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_PRESSURERISE)
  motor_efficiency = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_MOTOREFFICIENCY)

  # construct the fan
  fan = OpenStudio::Model::FanConstantVolume.new($osm.osm_model)
  fan.setPressureRise(pressure_rise) unless pressure_rise.nil?
  fan.setMotorEfficiency(motor_efficiency) unless motor_efficiency.nil?
  fan.setMaximumFlowRate(rated_flow_rate) unless rated_flow_rate.nil?

  # construct the gas heating coil
  coil = OpenStudio::Model::CoilHeatingGas.new($osm.osm_model)
  coil.setGasBurnerEfficiency(gas_efficiency) unless gas_efficiency.nil?
  coil.setNominalCapacity(rated_heat_cap) unless rated_heat_cap.nil?

  # the unit heater object requires an availability schedule, so just create a constant 1 schedule
  s = OpenStudio::Model::ScheduleConstant.new($osm.osm_model)
  s.setValue(1)

  # and finally the entire unit heater object
  $unit_heater = OpenStudio::Model::ZoneHVACUnitHeater.new($osm.osm_model, s, fan, coil)
  $unit_heater.setName(product_data.product_name)
  $unit_heater.setMaximumSupplyAirFlowRate(rated_flow_rate) unless rated_flow_rate.nil?
  $unit_heater.setFanControlType(fan_control) unless fan_control.nil?
end
