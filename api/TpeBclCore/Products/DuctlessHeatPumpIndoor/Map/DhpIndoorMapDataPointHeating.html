def initialize(outdoor_dry_bulb_temp, outdoor_wet_bulb_temp, indoor_dry_bulb_temp, heating_capacity, energy_input, avg_rated_heating_capacity)
  @outdoor_dry_bulb_temp = outdoor_dry_bulb_temp
  @outdoor_wet_bulb_temp = outdoor_wet_bulb_temp
  @indoor_dry_bulb_temp  = indoor_dry_bulb_temp
  @heating_capacity      = heating_capacity
  @heating_capacity_frac = if heating_capacity.nil?
    nil
  else
    heating_capacity / avg_rated_heating_capacity
                           end
  @energy_input = energy_input
end
