def initialize(outdoor_dry_bulb_temp, indoor_wet_bulb_temp, cooling_capacity, sensible_capacity, energy_input, avg_rated_cooling_capacity)
    @outdoor_dry_bulb_temp = outdoor_dry_bulb_temp
    @indoor_wet_bulb_temp  = indoor_wet_bulb_temp
    @cooling_capacity      = cooling_capacity
    if cooling_capacity.nil?
        @cooling_capacity_frac = nil
    else
        @cooling_capacity_frac = cooling_capacity / avg_rated_cooling_capacity
    end
    @sensible_capacity     = sensible_capacity
    @energy_input          = energy_input
end
