def self.get_bcl_parameters(product_data)
    parameters = {}
    params_to_include = [PARAM_RATEDCAPACITY, PARAM_ACTUALTANKVOLUME, PARAM_RATEDCOP, PARAM_RATEDAIRVOLUMEFLOW]
    params_to_include.each do |x|
        val = product_data.data_from_csv.get_parameter_average_value_by_descriptor(x)
        if val.nil?
            $log.log("Key BCL parameter, named '" + x + "', was not found in the input.  Skipping...")
        else
            parameters[x] = val
        end
    end
    return parameters
end

def self.suitable_for_bcl?(product_data)
    required_parameters = [PARAM_RATEDCAPACITY, PARAM_ACTUALTANKVOLUME, PARAM_RATEDCOP, PARAM_REFRIGERANT, PARAM_RATEDAIRVOLUMEFLOW]
    list_of_missing_parameters = []
    required_parameters.each do |param|
        if product_data.data_from_csv.get_parameter_average_value_by_descriptor(param).nil?
            list_of_missing_parameters.push(param)
        end
    end
    if list_of_missing_parameters.length.zero?
        return true
    else
        product_data.status_message = $log.issue_missing_parameter_error(list_of_missing_parameters)
        return false
    end
end

def self.write_base_idf_objects(product_data)

    # get parameters first
    rated_capacity = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_RATEDCAPACITY)
    if rated_capacity.nil?
        rated_capacity = false
        rated_capacity_string = ""
    else
        rated_capacity_string = rated_capacity.to_s
    end

    tank_volume = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_ACTUALTANKVOLUME)
    if tank_volume.nil?
        tank_volume = false
        tank_volume_string = ""
    else
        tank_volume_string = tank_volume.to_s
    end

    tank_height = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_STORAGETANKHEIGHT)
    if tank_height.nil?
        tank_height = false
        tank_height_string = ""
    else
        tank_height_string = tank_height.to_s
    end

    rated_cop = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_RATEDCOP)
    if rated_cop.nil?
        rated_cop = false
        rated_cop_string = ""
    else
        rated_cop_string = rated_cop.to_s
    end

    rated_shr = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_RATEDSHR)
    if rated_shr.nil?
        rated_shr = false
        rated_shr_string = ""
    else
        rated_shr_string = rated_shr.to_s
    end

    upper_htr_cap = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_UPPERHTRCAPACITY)
    if upper_htr_cap.nil?
        upper_htr_cap = false
        upper_htr_cap_string = ""
    else
        upper_htr_cap_string = upper_htr_cap.to_s
    end

    lower_htr_cap = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_LOWERHTRCAPACITY)
    if lower_htr_cap.nil?
        lower_htr_cap = false
        lower_htr_cap_string = ""
    else
        lower_htr_cap_string = lower_htr_cap.to_s
    end

    parasitic_rate = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_PARASITICRATE)
    if parasitic_rate.nil?
        parasitic_rate = false
        parasitic_rate_string = ""
    else
        parasitic_rate_string = parasitic_rate.to_s
    end

    rated_inlet_air_db = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_RATEDINLETAIRDB)
    if rated_inlet_air_db.nil?
        rated_inlet_air_db = false
        rated_inlet_air_db_string = ""
    else
        rated_inlet_air_db_string = rated_inlet_air_db.to_s
    end

    rated_inlet_air_wb = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_RATEDINLETAIRWB)
    if rated_inlet_air_wb.nil?
        rated_inlet_air_wb = false
        rated_inlet_air_wb_string = ""
    else
        rated_inlet_air_wb_string = rated_inlet_air_wb.to_s
    end

    rated_air_volume_flow = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_RATEDAIRVOLUMEFLOW)
    if rated_air_volume_flow.nil?
        rated_air_volume_flow = false
        rated_air_volume_flow_string = ""
    else
        rated_air_volume_flow_string = rated_air_volume_flow.to_s
    end

    off_cycle_heat_loss_coeff = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_OFFCYCLEHEATLOSSCOEFF)
    if off_cycle_heat_loss_coeff.nil?
        off_cycle_heat_loss_coeff = false
        off_cycle_heat_loss_coeff_string = ""
    else
        off_cycle_heat_loss_coeff_string = off_cycle_heat_loss_coeff.to_s
    end

    minimum_temp_for_compressor = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_MINIMUMAIRTEMPFORCOMP)
    if minimum_temp_for_compressor.nil?
        minimum_temp_for_compressor = false
        minimum_temp_for_compressor_string = ""
    else
        minimum_temp_for_compressor_string = minimum_temp_for_compressor.to_s
    end

    fan_total_efficiency = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_HEATPUMPFANTOTALEFFICIENCY)
    if fan_total_efficiency.nil?
        fan_total_efficiency = false
        fan_total_efficiency_string = ""
    else
        fan_total_efficiency = fan_total_efficiency / 100
        fan_total_efficiency_string = fan_total_efficiency.to_s
    end

    fan_rated_power_input = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_HEATPUMPFANRATEDPOWERINPUT)
    if fan_rated_power_input.nil?
        fan_rated_power_input = false
        fan_rated_power_input_string = ""
    else
        fan_rated_power_input_string = fan_rated_power_input.to_s
    end

    # calculate fan pressure rise if the other terms are available
    if !fan_rated_power_input or !fan_total_efficiency or !rated_air_volume_flow
        fan_pressure_rise = 1
        fan_pressure_rise_string = fan_pressure_rise.to_s
    else
        fan_pressure_rise = fan_rated_power_input * fan_total_efficiency / rated_air_volume_flow
        fan_pressure_rise_string = fan_pressure_rise.to_s
    end

    # now let's write the osc

    $water_tank = OpenStudio::Model::WaterHeaterStratified.new($osm.osm_model)
    # set water tank/heater parameters
    if tank_volume
        $water_tank.setTankVolume(tank_volume)
    end
    if tank_height
        $water_tank.setTankHeight(tank_height)
    end
    if upper_htr_cap
        $water_tank.setHeater1Capacity(upper_htr_cap)
    end
    if lower_htr_cap
        $water_tank.setHeater2Capacity(lower_htr_cap)
    end
    if off_cycle_heat_loss_coeff
        $water_tank.setUniformSkinLossCoefficientperUnitAreatoAmbientTemperature(off_cycle_heat_loss_coeff)
    end
    $water_tank.setTankShape("VerticalCylinder")
    $water_tank.setHeaterFuelType("Electricity")
    $water_tank.setAmbientTemperatureIndicator("Zone")

    $coil = OpenStudio::Model::CoilWaterHeatingAirToWaterHeatPump.new($osm.osm_model)
    # set coil parameters, will assign the curves later
    if rated_capacity
        $coil.setRatedHeatingCapacity(rated_capacity)
    end
    if rated_cop
        $coil.setRatedCOP(rated_cop)
    end
    if rated_shr
        $coil.setRatedSensibleHeatRatio(rated_shr)
    end
    if rated_inlet_air_db
        $coil.setRatedEvaporatorInletAirDryBulbTemperature(rated_inlet_air_db)
    end
    if rated_inlet_air_wb
        $coil.setRatedEvaporatorInletAirWetBulbTemperature(rated_inlet_air_wb)
    end
    if rated_air_volume_flow
        $coil.setRatedEvaporatorAirFlowRate(rated_air_volume_flow)
    end
    $coil.autosizeRatedCondenserWaterFlowRate()
    $coil.setEvaporatorFanPowerIncludedinRatedCOP(false)
    $coil.setCondenserPumpPowerIncludedinRatedCOP(false)
    $coil.setCondenserPumpHeatIncludedinRatedHeatingCapacityandRatedCOP(false)
    $coil.setEvaporatorAirTemperatureTypeforCurveObjects("WetBulbTemperature")

    $fan_onoff = OpenStudio::Model::FanOnOff.new($osm.osm_model)
    # set fan parameters
    if fan_total_efficiency
        $fan_onoff.setFanEfficiency(fan_total_efficiency)
    end
    if fan_pressure_rise
        $fan_onoff.setPressureRise(fan_pressure_rise)
    end
    if rated_air_volume_flow
        $fan_onoff.setMaximumFlowRate(rated_air_volume_flow)
    end

    # the heat pump water heater object requires two schedule objects (compressor setpoint temp and inlet air mixer)
    # for now just create a constant on schedule
    s = OpenStudio::Model::ScheduleConstant.new($osm.osm_model)
    s.setValue(1)

    $hpwh = OpenStudio::Model::WaterHeaterHeatPump.new($osm.osm_model, $coil, $water_tank, $fan_onoff, s, s)
    # set water heater heat pump parameters
    $hpwh.setName(product_data.product_name)
    if !minimum_temp_for_compressor
      # no min temp given...default
    elsif minimum_temp_for_compressor < 5
      $hpwh.setMinimumInletAirTemperatureforCompressorOperation(5)
    else
      $hpwh.setMinimumInletAirTemperatureforCompressorOperation(minimum_temp_for_compressor)
    end

    $idf.idf("

  WaterHeater:HeatPump,
    #{product_data.product_name},     !- Name
    ,                        !- Availability Schedule Name
    HPWHTempSch,             !- Compressor Setpoint Temperature Schedule Name
    ,                        !- Dead Band Temperature Difference {deltaC}
    ,                        !- Condenser Water Inlet Node Name
    ,                        !- Condenser Water Outlet Node Name
    autocalculate,           !- Condenser Water Flow Rate {m3/s}
    autocalculate,           !- Evaporator Air Flow Rate {m3/s}
    ZoneAirOnly,             !- Inlet Air Configuration
    ,                        !- Air Inlet Node Name
    ,                        !- Air Outlet Node Name
    ,                        !- Outdoor Air Node Name
    ,                        !- Exhaust Air Node Name
    ,                        !- Inlet Air Temperature Schedule Name
    ,                        !- Inlet Air Humidity Schedule Name
    ,                        !- Inlet Air Zone Name
    WaterHeater:Stratified,  !- Tank Object Type
    HPWHTank,                !- Tank Name
    ,                        !- Tank Use Side Inlet Node Name
    ,                        !- Tank Use Side Outlet Node Name
    Coil:WaterHeating:AirToWaterHeatPump,  !- DX Coil Object Type
    HPWHDXCoil,              !- DX Coil Name
    #{minimum_temp_for_compressor_string}, !- Minimum Inlet Air Temperature for Compressor Operation {C}
    Zone,                    !- Compressor Location
    ,                        !- Compressor Ambient Temperature Schedule Name
    Fan:OnOff,               !- Fan Object Type
    HPWHFan;                 !- Fan Object Name

  WaterHeater:Stratified,
    HPWHTank,                !- Name
    ,                        !- End-Use Subcategory
    #{tank_volume_string},   !- Tank Volume {m3}
    #{tank_height_string},   !- Tank Height {m}
    VerticalCylinder,        !- Tank Shape
    ,                        !- Tank Perimeter
    ,                        !- Maximum Temperature Limit
    ,                        !- Heater Priority Control
    HPWHTempSch,             !- Heater 1 Setpoint Temperature Schedule Name
    ,                        !- Heater 1 Deadband Temperature Difference {deltaC}
    #{upper_htr_cap_string}, !- Heater 1 Capacity
    ,                        !- Heater 1 Height
    HPWHTempSch,             !- Heater 2 Setpoint Temperature Schedule Name
    ,                        !- Heater 2 Deadband Temperature Difference {deltaC}
    #{lower_htr_cap_string}, !- Heater 2 Capacity
    ,                        !- Heater 2 Height
    Electricity,             !- Heater Fuel Type
    ,                        !- Heater Thermal Efficiency
    #{parasitic_rate_string}, !- Off Cycle Parasitic Fuel Consumption Rate {W}
    Electricity,             !- Off Cycle Parasitic Fuel Type
