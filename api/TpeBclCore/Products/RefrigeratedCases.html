def self.get_bcl_parameters(product_data)
  parameters = {}
  params_to_include = [PARAM_RATEDTOTALCOOLINGCAPACITY, PARAM_INTENDEDUSE, PARAM_TYPEOFREFRIGERATOR, PARAM_TEMPERATURECLASS]
  params_to_include.each do |x|
    parameters[x] = product_data.data_from_csv.get_parameter_average_value_by_descriptor(x)
  end
  parameters
end

def self.suitable_for_bcl?(product_data)
  return false if product_data.data_from_csv.nil?
  if product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_RATEDTOTALCOOLINGCAPACITY).nil?
    return false
  end
  if product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_CASELENGTH).nil?
    return false
  end
  if product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_RATEDEVAPORATORFANPOWER).nil?
    return false
  end

  if product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_DESIGNINLETEVAPORATORTEMPERATURE).nil?
    return false
  end
  this_refrigerator_type = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_TYPEOFREFRIGERATOR)
  if this_refrigerator_type.nil?
    return false
  elsif this_refrigerator_type == 'Vertical Closed Transparent (VCT)' || this_refrigerator_type == 'Horizontal Closed Transparent (HCT)'
    if product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_MAXIMUMANTISWEATDOORHEATERPOWER).nil? && product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_MAXIMUMFRAMEHEATERPOWER).nil?
      return false
    end
    # Check for other refrigerator type required values if needed
  end
  this_lighting_type = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_TYPEOFREFRIGERATORLIGHTING)
  if this_lighting_type.nil?
    return false
  elsif ['Incandescent', 'Fluorescent', 'LED', 'Other'].include?(this_lighting_type)
    if product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_RATEDCASELIGHTINGPOWER).nil?
      return false
    end
    # Check for other lighting type required values if needed
  end
  this_defrost_type = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_CASEDEFROSTTYPE)
  return false if this_defrost_type.nil?
  if ['Electric', 'Electric with Temperature Termination'].include?(this_defrost_type)
    if product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_CASEDEFROSTPOWER).nil?
      return false
    end
  end
  # If all tests pass, return true
  true
end

def self.write_component(product_data)
  case_watts = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_RATEDTOTALCOOLINGCAPACITY)
  case_length = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_CASELENGTH)
  case_safety = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_TOTALCOOLINGCAPACITYSAFETYFACTOR)
  case_latent = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_RATEDLATENTCASECAPACITY)
  case_rated_fan = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_RATEDEVAPORATORFANPOWER)
  case_installed_fan = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_INSTALLEDEVAPORATORFANPOWER)
  case_rated_light = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_RATEDCASELIGHTINGPOWER)
  case_installed_light = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_INSTALLEDCASELIGHTINGPOWER)
  case_door_heat = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_MAXIMUMANTISWEATDOORHEATERPOWER)
  case_frame_heat = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_MAXIMUMFRAMEHEATERPOWER)
  case_height = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_CASEHEIGHT)
  case_defrost_power = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_CASEDEFROSTPOWER)
  case_defrost_type = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_CASEDEFROSTTYPE)
  case_evap_temp = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_DESIGNINLETEVAPORATORTEMPERATURE)

  antisweat_heater_power = nil
  if (case_door_heat && case_length) && !case_frame_heat
    antisweat_heater_power = case_door_heat / case_length
  elsif (case_frame_heat && case_length) && !case_door_heat
    antisweat_heater_power = case_frame_heat / case_length
  elsif (case_door_heat && case_length) && (case_frame_heat && case_length)
    antisweat_heater_power = (case_frame_heat + case_door_heat) / case_length
  end

  # the refrigeration case object requires a schedule object, which defines the defrost operation
  # for now, I'm just creating a constant zero schedule which implies **no defrost**
  # while this isn't right, I don't have much else to go on.
  s = OpenStudio::Model::ScheduleConstant.new($osm.osm_model)

  $rcase = OpenStudio::Model::RefrigerationCase.new($osm.osm_model, s)
  $rcase.setName(product_data.product_name)
  $rcase.setRatedAmbientTemperature(23.9)
  $rcase.setRatedAmbientRelativeHumidity(55)
  $rcase.setRatedTotalCoolingCapacityperUnitLength(case_watts / case_length) unless case_watts.nil? || case_length.nil? || case_length == 0
  $rcase.setRatedLatentHeatRatio(case_latent / case_watts) unless case_latent.nil? || case_watts.nil? || case_watts == 0
  $rcase.setRatedRuntimeFraction((case_watts - case_safety) / case_watts) unless case_watts.nil? || case_safety.nil? || case_watts == 0
  $rcase.setCaseLength(case_length) unless case_length.nil?
  $rcase.setStandardCaseFanPowerperUnitLength(case_rated_fan / case_length) unless case_rated_fan.nil? || case_length.nil? || case_length == 0
  $rcase.setOperatingCaseFanPowerperUnitLength(case_installed_fan / case_length) unless case_installed_fan.nil? || case_length.nil? || case_length == 0
  $rcase.setStandardCaseLightingPowerperUnitLength(case_rated_light / case_length) unless case_rated_light.nil? || case_length.nil? || case_length == 0
  $rcase.setInstalledCaseLightingPowerperUnitLength(case_installed_light / case_length) unless case_installed_light.nil? || case_length.nil? || case_length == 0
  $rcase.setCaseAntiSweatHeaterPowerperUnitLength(antisweat_heater_power) unless antisweat_heater_power.nil?
  $rcase.setCaseHeight(case_height) unless case_height.nil?
  $rcase.setCaseDefrostPowerperUnitLength(case_defrost_power / case_length) unless case_defrost_power.nil? || case_length.nil? || case_length == 0
  $rcase.setCaseDefrostType(case_defrost_type) unless case_defrost_type.nil?
  $rcase.setDesignEvaporatorTemperatureorBrineInletTemperature(case_evap_temp) unless case_evap_temp.nil?
end
