def initialize(map_spreadsheet_object, _csv_data, map_file_name)
  # read in the meta data
  owner = map_spreadsheet_object.cell(3, 4)
  brand         = map_spreadsheet_object.cell(4, 4)
  product_line  = map_spreadsheet_object.cell(5, 4)
  model         = map_spreadsheet_object.cell(6, 4)

  # instantiate the base class with common data
  super(owner, brand, product_line, model)
  @type_clg_or_htg_or_defrost = nil

  # next we need to read in PV Module specific data
  # first instantiate the main data point array
  @this_data_set = []

  # set default reference conditions
  independent_vars = []
  independent_vars << 1000
  independent_vars << 25

  # set reading bounds and other parameters, and start reading data
  row_short_ckt_open_ckt       = 13
  row_max_power_point          = 25
  column_short_ckt_and_power   = 5
  column_open_ckt_and_voltage  = 12

  # read the dependent variables
  dependent_vars = []
  dependent_vars << map_spreadsheet_object.cell(row_short_ckt_open_ckt, column_short_ckt_and_power)
  dependent_vars << map_spreadsheet_object.cell(row_short_ckt_open_ckt, column_open_ckt_and_voltage)
  dependent_vars << map_spreadsheet_object.cell(row_max_power_point, column_short_ckt_and_power)
  dependent_vars << map_spreadsheet_object.cell(row_max_power_point, column_open_ckt_and_voltage)

  valid_point = true
  for item in dependent_vars
    if item.nil?
    # blank, leave it alone for now
    elsif !Numerics.parameter_is_numeric?(item)
      # non-numerics, skip it
      $log.log("Found non-numerics in map data: #{dependent_vars}")
      valid_point = false
      # good to go
    end
  end

  # calculate current at max power point
  dependent_vars << dependent_vars[2] / dependent_vars[3] unless dependent_vars[3].nil?

  # create a new data point
  this_data_point = PVModuleMapDataPoint.new(independent_vars, dependent_vars)

  # add it to the main dataset hash
  @this_data_set.push(this_data_point) if valid_point

  spew('') # print base map data
  PVModuleMapDataPoint.spew # print class-level header line
  @this_data_set.each(&:spew) # print all data points data

  # throw an error if no data at reference conditions
  return unless @this_data_set.length.to_f < 1
  $log.log("Insufficient data points found for this map file: #{map_file_name}")
  @insufficient_data_points = true

end
