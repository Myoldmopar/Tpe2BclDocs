def self.get_bcl_parameters(product_data)
    parameters = {}
    params_to_include = [PARAM_NOMCLGCAPACITY,
                         PARAM_CLGFANCONTROL,
                         PARAM_CLGNUMSPEEDS,
                         PARAM_HTGFANCONTROL,
                         PARAM_HTGNUMSPEEDS,
                         PARAM_COMPRESSORSTAGING,
                         PARAM_NUMCOOLINGSTAGES,
                         PARAM_HEATPUMP,
                         PARAM_STATICPRESSURE]  # Need to check which parameters BCL requires
    params_to_include.each do |x|
        val = product_data.data_from_csv.get_parameter_average_value_by_descriptor(x)
        if val.nil?
            $log.log("Key BCL parameter, named '" + x + "', was not found in the input.  Skipping...")
        else
            parameters[x] = val
        end
    end
    return parameters
end

def self.suitable_for_bcl?(product_data)
    required_parameters = [PARAM_NOMCLGCAPACITY,
                           PARAM_CLGFANCONTROL,
                           PARAM_CLGNUMSPEEDS,
                           PARAM_HTGFANCONTROL,
                           PARAM_HTGNUMSPEEDS,
                           PARAM_COMPRESSORSTAGING,
                           PARAM_HEATPUMP,
                           PARAM_STATICPRESSURE] # Need to check which parameters BCL requires
    list_of_missing_parameters = []
    required_parameters.each do |param|
        if product_data.data_from_csv.get_parameter_average_value_by_descriptor(param).nil?
            list_of_missing_parameters.push(param)
        end
    end
    if list_of_missing_parameters.length.zero?
        return true
    else
        product_data.status_message = $log.issue_missing_parameter_error(list_of_missing_parameters)
        return false
    end
end

def self.write_base_idf_objects(product_data, merged_hp_data_set, merged_clg_data_set, merged_gaselec_htg_data_set)

    # determine if multispeed heat pump is used
    this_is_heat_pump = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_HEATPUMP)

    # determine heating type
    type_of_heating = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_HEATINGTYPE)

    # so we now know what configuration we need to create; verify we have good enough data here
    # start with the heating data requirements
    skip_heating = false
    if this_is_heat_pump == "Yes" # if a heat pump, then we need heat pump data and optionally supplemental heating data
        if merged_hp_data_set.length == 0
            $log.log("CSV parameters suggest that this product is a heat pump, yet the heat pump performance dataset was empty.")
            $log.log("Ensure that if this unit is a heat pump that performance data is provided via the Heat Pump Performance Map spreadsheet template.")
            raise TpeBclException, "Heat Pump performance data expected, but not found"
        end
        if type_of_heating == "None"
            # then we don't expect any supplemental heating
        else
            # then we expect some gas/elec dataset to be included
            if merged_gaselec_htg_data_set.length == 0
                $log.log("CSV parameters suggest that this product is a heat pump and has either gas or electric supplemental heating.")
                $log.log("However, the gas/electric heating dataset was empty.  Ensure that if this has gas/electric supplemental heating,")
                $log.log(" that the performance data is provided via the Gas or Electric Heat Performance Map spreadsheet template.")
                raise TpeBclException, "Gas/Electric Supplemental heating performance data expected, but not found"
            end
        end
    else # if not a heat pump then if there is heating, we need gas/elec data
        if type_of_heating == "None"
            # then we don't need any heating data
        else
            # then we expect some gas/elec dataset to be included
            if merged_gaselec_htg_data_set.length == 0
                $log.log("CSV parameters suggest that this product has a primary gas or electric heating coil.")
                $log.log("However, the gas/electric heating dataset was empty.  Ensure that if this has gas/electric heating,")
                $log.log(" that the performance data is provided via the Gas or Electric Heat Performance Map spreadsheet template.")
                $log.log("This translation will continue, but there will be no heating coil on the RTU.")
                skip_heating = true
            end
        end
    end
    # and then cooling, for now I'll make it always required
    if merged_clg_data_set.length == 0
        $log.log("Rooftop unit cooling dataset was empty.  Currently, cooling data is required.")
        $log.log("Ensure that cooling performance data is provided via the Cooling Performance Map spreadsheet template.")
        raise TpeBclException, "RTU Cooling performance data expected, but not found"
    end

    # get data at rated conditions: this is currently just the max flow rates for each fan speed
    number_of_htg_speeds = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_HTGNUMSPEEDS)
    number_of_clg_speeds = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_CLGNUMSPEEDS)

    # figure out which heating dataset we want to use for the primary coil
    nom_htg_flow_rate = -999 # initialize this to a negative value so that in cases with no heating, the cooling will always be higher
    if not skip_heating
        if this_is_heat_pump == "Yes"
            heating_data_set = merged_hp_data_set
        else
            heating_data_set = merged_gaselec_htg_data_set
        end
        if number_of_htg_speeds >= 1
            stage = 1
            datapts_speed1 = heating_data_set.select { |pnt| pnt.fan_speed_stage == stage}
            max_flow_point_htg = datapts_speed1.max_by { |pnt| pnt.supply_fan_flow_rate }
            htg_supply_air_flow_rate_1 = max_flow_point_htg.supply_fan_flow_rate
        end
        if number_of_htg_speeds == 1
            htg_supply_air_flow_rate_2 = htg_supply_air_flow_rate_1
            nom_htg_flow_rate = htg_supply_air_flow_rate_2
        end
        if number_of_htg_speeds >= 2
            stage = 2
            datapts_speed1 = heating_data_set.select { |pnt| pnt.fan_speed_stage == stage}
            max_flow_point_htg = datapts_speed1.max_by { |pnt| pnt.supply_fan_flow_rate }
            htg_supply_air_flow_rate_2 = max_flow_point_htg.supply_fan_flow_rate
            nom_htg_flow_rate = htg_supply_air_flow_rate_2
        end
        if number_of_htg_speeds >= 3
            stage = 3
            datapts_speed1 = heating_data_set.select { |pnt| pnt.fan_speed_stage == stage}
            max_flow_point_htg = datapts_speed1.max_by { |pnt| pnt.supply_fan_flow_rate }
            htg_supply_air_flow_rate_3 = max_flow_point_htg.supply_fan_flow_rate
            nom_htg_flow_rate = htg_supply_air_flow_rate_3
        end
        if number_of_htg_speeds >= 4
            stage = 4
            datapts_speed1 = heating_data_set.select { |pnt| pnt.fan_speed_stage == stage}
            max_flow_point_htg = datapts_speed1.max_by { |pnt| pnt.supply_fan_flow_rate }
            htg_supply_air_flow_rate_4 = max_flow_point_htg.supply_fan_flow_rate
            nom_htg_flow_rate = htg_supply_air_flow_rate_4
        end
    end

    # then cooling; if nom_htg_flow_rate is positive, it must have a valid max_flow_point_htg by now from above
    if number_of_clg_speeds >= 1
        stage = 1
        datapts_speed1 = merged_clg_data_set.select { |pnt| pnt.fan_speed_stage == stage}
        max_flow_point_clg = datapts_speed1.max_by { |pnt| pnt.supply_fan_flow_rate }
        clg_supply_air_flow_rate_1 = max_flow_point_clg.supply_fan_flow_rate
    end
    if number_of_clg_speeds == 1
        clg_supply_air_flow_rate_2 = clg_supply_air_flow_rate_1
        nom_clg_flow_rate = clg_supply_air_flow_rate_2
        if nom_clg_flow_rate > nom_htg_flow_rate
            fan_max_data_point = max_flow_point_clg
        else
            fan_max_data_point = max_flow_point_htg
        end
    end
    if number_of_clg_speeds >= 2
        stage = 2
        datapts_speed1 = merged_clg_data_set.select { |pnt| pnt.fan_speed_stage == stage}
        max_flow_point_clg = datapts_speed1.max_by { |pnt| pnt.supply_fan_flow_rate }
        clg_supply_air_flow_rate_2 = max_flow_point_clg.supply_fan_flow_rate
        nom_clg_flow_rate = clg_supply_air_flow_rate_2
        if nom_clg_flow_rate > nom_htg_flow_rate
            fan_max_data_point = max_flow_point_clg
        else
            fan_max_data_point = max_flow_point_htg
        end
    end
    if number_of_clg_speeds >= 3
        stage = 3
        datapts_speed1 = merged_clg_data_set.select { |pnt| pnt.fan_speed_stage == stage}
        max_flow_point_clg = datapts_speed1.max_by { |pnt| pnt.supply_fan_flow_rate }
        clg_supply_air_flow_rate_3 = max_flow_point_clg.supply_fan_flow_rate
        nom_clg_flow_rate = clg_supply_air_flow_rate_3
        if nom_clg_flow_rate > nom_htg_flow_rate
            fan_max_data_point = max_flow_point_clg
        else
            fan_max_data_point = max_flow_point_htg
        end
    end
    if number_of_clg_speeds >= 4
        stage = 4
        datapts_speed1 = merged_clg_data_set.select { |pnt| pnt.fan_speed_stage == stage}
        max_flow_point_clg = datapts_speed1.max_by { |pnt| pnt.supply_fan_flow_rate }
        clg_supply_air_flow_rate_4 = max_flow_point_clg.supply_fan_flow_rate
        nom_clg_flow_rate = clg_supply_air_flow_rate_4
        if nom_clg_flow_rate > nom_htg_flow_rate
            fan_max_data_point = max_flow_point_clg
        else
            fan_max_data_point = max_flow_point_htg
        end
    end

    # get fan data
    fan_max_flow_rate     = fan_max_data_point.supply_fan_flow_rate
    fan_delta_pressure    = fan_max_data_point.total_static_pressure
    fan_mechanical_power  = fan_max_flow_rate * fan_delta_pressure
    fan_total_power_input = fan_max_data_point.supply_fan_power_draw
    fan_total_efficiency  = fan_mechanical_power / fan_total_power_input unless fan_total_power_input.nil?

    # create new roof top unit object from the model
    $rtu = OpenStudio::Model::AirLoopHVACUnitarySystem.new($osm.osm_model)
    $rtu.setName(product_data.product_name)

    # create fan object
    my_fan = OpenStudio::Model::FanOnOff.new($osm.osm_model)
    # set os fan params
    my_fan.setFanEfficiency(fan_total_efficiency)
    my_fan.setPressureRise(fan_delta_pressure)
    my_fan.setMaximumFlowRate(fan_max_flow_rate)

    # set rtu fan and air flow rates
    $rtu.setSupplyFan(my_fan)
    $rtu.setSupplyAirFlowRateMethodDuringCoolingOperation("SupplyAirFlowRate")
    $rtu.setSupplyAirFlowRateDuringCoolingOperation(nom_clg_flow_rate)
    $rtu.setSupplyAirFlowRateMethodDuringHeatingOperation("SupplyAirFlowRate")
    $rtu.setSupplyAirFlowRateDuringHeatingOperation(nom_htg_flow_rate)

end

def self.write_idf_stuff(product_data)

    # merge data sets
    merged_hp_data_set      = TpeBclCore::MapMethods.merge_multiple_maps(product_data.data_from_maps, "heat pump", TpeBclCore::Products::RoofTopUnit::Map::RTUMapDataPointHP)
    merged_clg_data_set     = TpeBclCore::MapMethods.merge_multiple_maps(product_data.data_from_maps, "cooling", TpeBclCore::Products::RoofTopUnit::Map::RTUMapDataPointClg)
    merged_gaselec_htg_data_set = TpeBclCore::MapMethods.merge_multiple_maps(product_data.data_from_maps, "gaselec heating", TpeBclCore::Products::RoofTopUnit::Map::RTUMapDataPointElecGasHtg)

    # write base idf -- creates objects for AirLoopHVAC:UnitarySystem and Fan:OnOff
    write_base_idf_objects(product_data, merged_hp_data_set, merged_clg_data_set, merged_gaselec_htg_data_set)

    # write map idf -- creates objects for UnitarySystemPerformance:HeatPump:MultiSpeed, Coil:Cooling:DX:MultiSpeed, Coil:Heating:DX:MultiSpeed, Coil:Heating:Gas, Coil:Heating:Electric
    write_map_idf_objects(product_data, merged_hp_data_set, "heat pump")
    write_map_idf_objects(product_data, merged_clg_data_set, "cooling")
    write_map_idf_objects(product_data, merged_gaselec_htg_data_set, "gaselec heating")

end
