def self.get_bcl_parameters(product_data)
  parameters = {}
  params_to_include = [PARAM_RATEDFLOWRATE,
                       PARAM_RATEDPUMPHEAD,
                       PARAM_RATEDMOTORPOWER]
  params_to_include.each do |x|
    val = product_data.data_from_csv.get_parameter_average_value_by_descriptor(x)
    if val.nil?
      $log.log("Key BCL parameter, named '" + x + "', was not found in the input.  Skipping...")
    else
      parameters[x] = val
    end
  end
  parameters
end

def self.write_component(product_data)
  rated_flow_rate = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_RATEDFLOWRATE)
  rated_pump_head = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_RATEDPUMPHEAD)
  rated_power_consumption = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_RATEDMOTORPOWER)

  $pump = OpenStudio::Model::PumpVariableSpeed.new($osm.osm_model)
  $pump.setname(product_data.product_name)
  $pump.setRatedFlowRate(rated_flow_rate) unless rated_flow_rate.nil?
  $pump.setRatedPumpHead(rated_pump_head) unless rated_pump_head.nil?
  $pump.setRatedPowerConsumption(rated_power_consumption) unless rated_power_consumption.nil?
end
