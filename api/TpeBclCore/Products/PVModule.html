def self.get_bcl_parameters(product_data)
    parameters = {}
    params_to_include = [PARAM_CELLTYPE,
                         PARAM_NUMBEROFCELLS,
                         PARAM_SOLARCELLAREA,
                         PARAM_NOMOPERATINGCELLTEMP,
                         PARAM_SHORTCKTCURRENTTEMPCOEF,
                         PARAM_OPENCKTVOLTAGETEMPCOEF]
    params_to_include.each do |x|
        val = product_data.data_from_csv.get_parameter_average_value_by_descriptor(x)
        if val.nil?
            $log.log("Key BCL parameter, named '" + x + "', was not found in the input.  Skipping...")
        else
            parameters[x] = val
        end
    end
    return parameters
end

def self.suitable_for_bcl?(product_data)
    required_parameters = [PARAM_SOLARCELLAREA]
    list_of_missing_parameters = []
    required_parameters.each do |param|
        if product_data.data_from_csv.get_parameter_average_value_by_descriptor(param).nil?
            list_of_missing_parameters.push(param)
        end
    end
    if list_of_missing_parameters.length.zero?
        return true
    else
        $log.log("This product is missing key performance parameter definitions which make it unsuitable for pushing to the BCL.")
        $log.log("Missing parameters include:")
        list_of_missing_parameters.each do |param|
            $log.log("  #{param}")
        end
        return false
    end
end

def self.write_map_idf_objects(product_data, merged_data_set, type)

    # get parameters first
    module_efficiency = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_MODULEEFFICIENCY)
    if module_efficiency.nil?
        module_efficiency = false
        module_efficiency_string = ""
    else
        module_efficiency_string = module_efficiency.to_s
    end

    rated_power = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_RATEDPOWER)
    if rated_power.nil?
        rated_power = false
        rated_power_string = ""
    else
        rated_power_string = rated_power.to_s
    end

    cell_type = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_CELLTYPE)
    if cell_type.nil?
        cell_type = false
        cell_type_string = ""
    else
        cell_type_string = cell_type
    end

    number_of_cells = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_NUMBEROFCELLS)
    if number_of_cells.nil?
        number_of_cells = false
        number_of_cells_string = ""
    else
        number_of_cells_string = number_of_cells.to_s
    end

    solar_cell_area_sqcm = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_SOLARCELLAREA)
    solar_cell_area = solar_cell_area_sqcm*0.0001*number_of_cells
    if solar_cell_area.nil?
        solar_cell_area = false
        solar_cell_area_string = ""
    else
        solar_cell_area_string = solar_cell_area.to_s
    end

    nom_operating_cell_temp = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_NOMOPERATINGCELLTEMP)
    if nom_operating_cell_temp.nil?
        nom_operating_cell_temp = false
        nom_operating_cell_temp_string = ""
    else
        nom_operating_cell_temp_string = nom_operating_cell_temp.to_s
    end

    max_power_temp_coef = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_MAXPOWERTEMPCOEF)
    if max_power_temp_coef.nil?
        max_power_temp_coef = false
        max_power_temp_coef_string = ""
    else
        max_power_temp_coef_string = max_power_temp_coef.to_s
    end

    if merged_data_set[0].nil?
        short_ckt_current, open_ckt_voltage, current_at_max_power, voltage_at_max_power = "","","",""
    else
        short_ckt_current    = merged_data_set[0].short_ckt_current
        open_ckt_voltage     = merged_data_set[0].open_ckt_voltage
        current_at_max_power = merged_data_set[0].current_at_max_power
        voltage_at_max_power = merged_data_set[0].voltage_at_max_power
    end

    short_ckt_current_temp_coef = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_SHORTCKTCURRENTTEMPCOEF)
    if short_ckt_current_temp_coef.nil?
      short_ckt_current_temp_coef = false
      short_ckt_current_temp_coef_string = ""
    else
      short_ckt_current_temp_coef_string = (0.01 * short_ckt_current_temp_coef * short_ckt_current).to_s
    end

    open_ckt_voltage_temp_coef = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_OPENCKTVOLTAGETEMPCOEF)
    if open_ckt_voltage_temp_coef.nil?
      open_ckt_voltage_temp_coef = false
      open_ckt_voltage_temp_coef_string = ""
    else
      open_ckt_voltage_temp_coef_string = (0.01 * open_ckt_voltage_temp_coef * open_ckt_voltage).to_s
    end

    $idf.idf("

  PhotovoltaicPerformance:EquivalentOne-Diode,
    #{product_data.product_name},           !- Name
    #{cell_type_string},                    !- cell Type
    #{number_of_cells_string},              !- cells in series [-]
    #{solar_cell_area_string},              !- module area [m2]
    ,                                       !- tau alpha product at normal incidence [-]
    ,                                       !- electron bandgap [eV]
    ,                                       !- shunt resistance [ohms]
    #{short_ckt_current.to_s},              !- short circuit current at reference conditions [A]
    #{open_ckt_voltage.to_s},               !- open circuit voltage at reference conditions [V]
    25,                                     !- temperature at reference conditions [C]
    1000,                                   !- radiation at reference conditions [W/m2]
    #{current_at_max_power.to_s},           !- current at max power [A]
    #{voltage_at_max_power.to_s},           !- voltage at max power [V]
    #{short_ckt_current_temp_coef_string},  !- temperature coefficient of short circuit current [A/K]
    #{open_ckt_voltage_temp_coef_string},   !- temperature coefficient of open circuit voltage [V/K]
    ,                                       !- ambient temperature at NOCT [C]
    #{nom_operating_cell_temp_string},      !- cell temperature at NOCT [C]
    ,                                       !- radiation at NOCT [W/m2]
    ,                                       !- heat loss coefficient [W/m2.K]
    ;                                       !- total heat capacity (only used in TC mode 1)
    ")

end
