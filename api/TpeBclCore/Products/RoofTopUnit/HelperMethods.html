def self.get_gaselec_htg_rated_data_points(product_data, merged_gaselec_htg_data_set)
  # set the rated conditions
  rated_mixed_entering_db_temp = -6.667
  # find out number of speeds
  heating_staging = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_HEATINGSTAGING)
  number_of_heating_stages = product_data.data_from_csv.get_parameter_average_value_by_descriptor(PARAM_NUMHEATINGSTAGES)
  if heating_staging.nil?
    number_of_heating_stages = 0
  elsif heating_staging == 'Single Stage'
    number_of_heating_stages = 1
    # number of heating stages = number of heating stages
  end

  rated_gaselec_htg_data_points = {}

  if number_of_heating_stages >= 1
    # create array of speed 1 data points
    gaselec_htg_data_set_speed_1 = merged_gaselec_htg_data_set.select { |pnt| pnt.heating_stage == 1 }
    # pick out data points at rated conditions
    gaselec_htg_data_point_at_rated_conditions_1 = gaselec_htg_data_set_speed_1.select { |pnt| pnt.mixed_entering_db_temp.round(3) == rated_mixed_entering_db_temp }
    # add data points to correct array in hash
    rated_gaselec_htg_data_points['stage1'] = gaselec_htg_data_point_at_rated_conditions_1
  end
  if number_of_heating_stages == 1
    # add speed 1 rated data points to speed 2
    rated_gaselec_htg_data_points['stage2'] = gaselec_htg_data_point_at_rated_conditions_1
  end
  if number_of_heating_stages >= 2
    # create array of speed 2 data points
    gaselec_htg_data_set_speed_2 = merged_gaselec_htg_data_set.select { |pnt| pnt.heating_stage == 2 }
    # pick out data points at rated conditions
    gaselec_htg_data_point_at_rated_conditions_2 = gaselec_htg_data_set_speed_2.select { |pnt| pnt.mixed_entering_db_temp.round(3) == rated_mixed_entering_db_temp }
    # add data points to correct array in hash
    rated_gaselec_htg_data_points['stage2'] = gaselec_htg_data_point_at_rated_conditions_2
  end
  if number_of_heating_stages >= 3
    # create array of speed 3 data points
    gaselec_htg_data_set_speed_3 = merged_gaselec_htg_data_set.select { |pnt| pnt.heating_stage == 3 }
    # pick out data points at rated conditions
    gaselec_htg_data_point_at_rated_conditions_3 = gaselec_htg_data_set_speed_3.select { |pnt| pnt.mixed_entering_db_temp.round(3) == rated_mixed_entering_db_temp }
    # add data points to correct array in hash
    rated_gaselec_htg_data_points['stage3'] = gaselec_htg_data_point_at_rated_conditions_3
  end
  if number_of_heating_stages >= 4
    # create array of speed 4 data points
    gaselec_htg_data_set_speed_4 = merged_gaselec_htg_data_set.select { |pnt| pnt.heating_stage == 4 }
    # pick out data points at rated conditions
    gaselec_htg_data_point_at_rated_conditions_4 = gaselec_htg_data_set_speed_4.select { |pnt| pnt.mixed_entering_db_temp.round(3) == rated_mixed_entering_db_temp }
    # add data points to correct array in hash
    rated_gaselec_htg_data_points['stage4'] = gaselec_htg_data_point_at_rated_conditions_4
  end

  rated_gaselec_htg_data_points
end

def self.make_cooling_curves(data_for_func_flow_curve, data_for_func_temp_curve, cool_cap_x, cool_cop_x, rated_flow_rate)
  # pick out data points at highest fan speed for function of temperature curves
  this_cc_ft_set = []
  this_eir_ft_set = []
  data_for_func_temp_curve.each do |data_point|
    # create set of cooling capacities
    cooling_cap_frac = data_point.gross_total_clg_cap / cool_cap_x # normalize cooling capacity
    this_cc_ft_set.push([data_point.evap_entering_wb_temp, data_point.cond_entering_db_wb_temp, cooling_cap_frac])
    # create set of energy input ratios
    energy_input_ratio_frac = cool_cop_x * (data_point.gross_power_draw - data_point.supply_fan_power_draw) / data_point.gross_total_clg_cap # normalize EIR
    this_eir_ft_set.push([data_point.evap_entering_wb_temp, data_point.cond_entering_db_wb_temp, energy_input_ratio_frac])
  end

  # pick out flow fractions for each fan speed
  this_cc_ff_set = []
  this_eir_ff_set = []
  unique_fan_speeds = []
  data_for_func_flow_curve.each do |data_point|
    flow_fraction = data_point.supply_fan_flow_rate / rated_flow_rate
    # create set of cooling capacities
    cooling_cap_frac = data_point.gross_total_clg_cap / cool_cap_x # normalize cooling capacity
    this_cc_ff_set << [flow_fraction, cooling_cap_frac]
    # create set of energy input ratios
    energy_input_ratio_frac = cool_cop_x * (data_point.gross_power_draw - data_point.supply_fan_power_draw) / data_point.gross_total_clg_cap # normalize EIR
    this_eir_ff_set << [flow_fraction, energy_input_ratio_frac]
  end

  stage_curves = []

  # HPACCoolCapFT Speed x,           !- Total Cooling Capacity Function of Temperature Curve Name
  # TCC = f(T_coil_wb, T_condenser_db_wb)
  if this_cc_ft_set.any?
    unit_types = [OSMWriter::UNIT_TYPE_TEMPERATURE, OSMWriter::UNIT_TYPE_TEMPERATURE, OSMWriter::UNIT_TYPE_DIMENSIONLESS]
    ccft_table_curve = $osm.write_two_indep_var_table(this_cc_ft_set, unit_types)
    stage_curves[0] = ccft_table_curve
  else
    coefficients = [1, 0, 0]
    x_bounds = [0, 1]
    y_bounds = [0, 1]
    stage_curves[0] = $osm.write_quadratic_curve(coefficients, x_bounds, y_bounds, unit_types)
  end

  # HPACCoolCapFF Speed x,           !- Total Cooling Capacity Function of Flow Fraction Curve Name
  # TCC = f(flow_fraction)
  if this_cc_ff_set.any?
    unit_types = [OSMWriter::UNIT_TYPE_DIMENSIONLESS, OSMWriter::UNIT_TYPE_DIMENSIONLESS]
    ccff_table_curve = $osm.write_one_indep_var_table(this_cc_ff_set, unit_types, 'Quadratic')
    stage_curves[1] = ccff_table_curve
  else
    coefficients = [1, 0, 0]
    x_bounds = [0, 1]
    y_bounds = [0, 1]
    stage_curves[1] = $osm.write_quadratic_curve(coefficients, x_bounds, y_bounds, unit_types)
  end

  # HPACCoolEIRFT Speed x,           !- Energy Input Ratio Function of Temperature Curve Name
  # EIR = f(T_coil_wb, T_condenser_db_wb)
  if this_eir_ft_set.any?
    unit_types = [OSMWriter::UNIT_TYPE_TEMPERATURE, OSMWriter::UNIT_TYPE_TEMPERATURE, OSMWriter::UNIT_TYPE_DIMENSIONLESS]
    eirft_table_curve = $osm.write_two_indep_var_table(this_eir_ft_set, unit_types)
    stage_curves[2] = eirft_table_curve
  else
    coefficients = [1, 0, 0]
    x_bounds = [0, 1]
    y_bounds = [0, 1]
    stage_curves[2] = $osm.write_quadratic_curve(coefficients, x_bounds, y_bounds, unit_types)
  end

  # HPACCoolEIRFF Speed x,           !- Energy Input Ratio Function of FLow Fraction Curve Name
  # EIR = f(flow_fraction)
  if this_eir_ff_set.any?
    unit_types = [OSMWriter::UNIT_TYPE_DIMENSIONLESS, OSMWriter::UNIT_TYPE_DIMENSIONLESS]
    eirff_table_curve = $osm.write_one_indep_var_table(this_eir_ff_set, unit_types, 'Quadratic')
    stage_curves[3] = eirff_table_curve
  else
    unit_types = [OSMWriter::UNIT_TYPE_DIMENSIONLESS, OSMWriter::UNIT_TYPE_DIMENSIONLESS]
    coefficients = [1, 0, 0]
    x_bounds = [0, 1]
    y_bounds = [0, 1]
    stage_curves[3] = $osm.write_quadratic_curve(coefficients, x_bounds, y_bounds, unit_types)
  end

  # HPACCoolPLFFPLR Speed x,         !- Part Load Fraction Function of Part Load Ratio
  # PLF = f(PLR)
  coefficients = [0.85, 0.15, 0]
  x_bounds = [0, 1]
  y_bounds = [0.7, 1]
  unit_types = [OSMWriter::UNIT_TYPE_DIMENSIONLESS, OSMWriter::UNIT_TYPE_DIMENSIONLESS]

  plf_curve = $osm.write_quadratic_curve(coefficients, x_bounds, y_bounds, unit_types)
  stage_curves[4] = plf_curve

  # stage data requires waste heat function of temp curve..
  unit_types = [OSMWriter::UNIT_TYPE_TEMPERATURE, OSMWriter::UNIT_TYPE_TEMPERATURE, OSMWriter::UNIT_TYPE_DIMENSIONLESS]
  waste_heat_recovery = $osm.write_two_indep_var_table([[0, 0, 1], [0, 0, 1]], unit_types)
  stage_curves[5] = waste_heat_recovery

  stage_curves
end

def self.make_heatpump_curves(data_for_func_flow_curve, data_for_func_temp_curve, heat_cap_x, heat_cop_x, rated_flow_rate)
  # pick out data points at highest fan speed for function of temperature curves
  this_hc_ft_set = []
  this_eir_ft_set = []
  data_for_func_temp_curve.each do |data_point|
    # create set of heating capacities
    heating_cap_frac = data_point.gross_total_htg_cap / heat_cap_x # normalize heating capacity
    this_hc_ft_set.push([data_point.indr_coil_entering_db_temp, data_point.outdr_coil_entering_db_temp, heating_cap_frac])
    # create set of energy input ratios
    energy_input_ratio_frac = heat_cop_x * (data_point.gross_power_draw - data_point.supply_fan_power_draw) / data_point.gross_total_htg_cap
    this_eir_ft_set.push([data_point.indr_coil_entering_db_temp, data_point.outdr_coil_entering_db_temp, energy_input_ratio_frac])
  end

  # pick out flow fractions for each fan speed
  this_hc_ff_set = []
  this_eir_ff_set = []
  data_for_func_flow_curve.each do |data_point|
    flow_fraction = data_point.supply_fan_flow_rate / rated_flow_rate
    # create set of heating capacities
    heating_cap_frac = data_point.gross_total_htg_cap / heat_cap_x # normalize cooling capacity
    this_hc_ff_set << [flow_fraction, heating_cap_frac]
    # create set of energy input ratios
    energy_input_ratio_frac = heat_cop_x * (data_point.gross_power_draw - data_point.supply_fan_power_draw) / data_point.gross_total_htg_cap # normalize EIR
    this_eir_ff_set << [flow_fraction, energy_input_ratio_frac]
  end

  stage_curves = []

  # HPACHeatCapFT Speed x,           !- Total Heating Capacity Function of Temperature Curve Name
  # THC = f(T_indoor_db, T_outdoor_db)
  if this_hc_ft_set.any?
    unit_types = [OSMWriter::UNIT_TYPE_TEMPERATURE, OSMWriter::UNIT_TYPE_TEMPERATURE, OSMWriter::UNIT_TYPE_DIMENSIONLESS]
    hcft_table_curve = $osm.write_two_indep_var_table(this_hc_ft_set, unit_types)
    stage_curves[0] = hcft_table_curve
  end

  # HPACHeatCapFF Speed x,           !- Total Heating Capacity Function of Flow Fraction Curve Name
  # THC = f(flow_fraction)
  if this_hc_ff_set.any?
    unit_types = [OSMWriter::UNIT_TYPE_DIMENSIONLESS, OSMWriter::UNIT_TYPE_DIMENSIONLESS]
    hcff_table_curve = $osm.write_one_indep_var_table(this_hc_ff_set, unit_types, 'Quadratic')
    stage_curves[1] = hcff_table_curve
  end

  # HPACHeatEIRFT Speed x,           !- Energy Input Ratio Function of Temperature Curve Name
  # EIR = f(T_indoor_db, T_outdoor_db)
  if this_eir_ft_set.any?
    unit_types = [OSMWriter::UNIT_TYPE_TEMPERATURE, OSMWriter::UNIT_TYPE_TEMPERATURE, OSMWriter::UNIT_TYPE_DIMENSIONLESS]
    eirft_table_curve = $osm.write_two_indep_var_table(this_eir_ft_set, unit_types)
    stage_curves[2] = eirft_table_curve
  end

  # HPACHeatEIRFF Speed x,           !- Energy Input Ratio Function of FLow Fraction Curve Name
  # EIR = f(flow_fraction)
  if this_eir_ff_set.any?
    unit_types = [OSMWriter::UNIT_TYPE_DIMENSIONLESS, OSMWriter::UNIT_TYPE_DIMENSIONLESS]
    eirff_table_curve = $osm.write_one_indep_var_table(this_eir_ff_set, unit_types, 'Quadratic')
    stage_curves[3] = eirff_table_curve
  end

  # HPACHeatPLFFPLR Speed x,         !- Part Load Fraction Function of Part Load Ratio
  # PLF = f(PLR)
  coefficients = [0.85, 0.15, 0]
  x_bounds = [0, 1]
  y_bounds = [0.7, 1]
  unit_types = [OSMWriter::UNIT_TYPE_DIMENSIONLESS, OSMWriter::UNIT_TYPE_DIMENSIONLESS]

  plf_curve = $osm.write_quadratic_curve(coefficients, x_bounds, y_bounds, unit_types)
  stage_curves[4] = plf_curve

  # stage data requires waste heat function of temp curve..
  unit_types = [OSMWriter::UNIT_TYPE_TEMPERATURE, OSMWriter::UNIT_TYPE_TEMPERATURE, OSMWriter::UNIT_TYPE_DIMENSIONLESS]
  waste_heat_recovery = $osm.write_two_indep_var_table([[0, 0, 1], [0, 0, 1]], unit_types)
  stage_curves[5] = waste_heat_recovery

  stage_curves
