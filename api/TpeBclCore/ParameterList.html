def get_parameter_average_value_by_descriptor(descriptor)
    $log.log("Calculating an average value for parameter: #{descriptor}")
    out_val = 0
    params = get_parameters_by_descriptor(descriptor)
    if params.length.zero?
        $log.log("No parameters found for this parameter.  Returning nil.")
        return nil
    end
    # need to split out here to allow for numeric/non-numeric parameters to be "averaged"; just check param index 0
    param_one_value = params[0].param_value
    param_one_value_no_spaces = param_one_value.gsub!(" ", "")
    if Numerics::parameter_is_numeric?(param_one_value)
        out_val = numeric_average(params)
    elsif Numerics::parameter_is_numeric?(param_one_value_no_spaces)
        out_val = numeric_average(params, true)
    else
        out_val = string_average(params)
    end
    $log.log("Resulting average value = #{out_val.to_s}")
    return out_val
end

def numeric_average(parameter_list, cut_spaces = false)
    numerator, denominator = 0, 0
    parameter_list.each do |param|
        $log.log("Encountered parameter from source: #{param.param_source.type.to_s}/#{param.param_source.name.to_s}; value = #{param.param_value.to_s}")
        this_param_value = param.param_value
        if cut_spaces
            this_param_value.gsub!(" ", "")
            $log.log("Parameter value had space characters scrubbed")
        end
        weight = DataSource::WEIGHTBYSOURCETYPE[param.param_source.type]
        numerator += weight * this_param_value.to_f
        denominator += weight
    end
    return numerator/denominator
end

def string_average(parameter_list)

    # precedence:
    #  - if only 1 parameter, it wins
    #  - third party is highest precendence, followed by contributor, followed by manufacturer
    #  - if multiple parameters of the same type exist the last one wins

    # first create a hash to map types to scores
    score_by_source = {DataSource::SOURCE_THIRDPARTYLAB => 3, DataSource::SOURCE_EVALUATOR => 2, DataSource::SOURCE_MANUFACTURER => 1}

    # initialize variables here for broader scope
    winning_parameter = nil
    winning_score = nil

    # now iterate
    parameter_list.each_with_index do |param, i|
        # always take the first item
        if i.zero?
            winning_parameter = parameter_list.first
            winning_score = score_by_source[winning_parameter.param_source.type]
            next
        end
        # but check each one afterwards
        this_param_score = score_by_source[param.param_source.type]
        if this_param_score >= winning_score
            winning_parameter = param
            winning_score = this_param_score
        end
    end

    # return the winner
    return winning_parameter.param_value

end
