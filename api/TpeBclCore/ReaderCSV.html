def self.get_csv_parameters(file)
  # initialize a list of parameters found in this csv file
  parameters = []

  # initialize the counter, this counter could be useful for reporting issues on a particular row
  csv_var_counter = -1

  # loop over all rows in the csv file
  CSV.foreach(file, encoding: 'ISO8859-1') do |row|
    # increment the counter
    csv_var_counter += 1

    # skip the first row (header)
    next if csv_var_counter.zero?

    # grab the information from the row tokens
    descriptor = row[0]
    if !descriptor.nil? && (descriptor.start_with?(' ') || descriptor.end_with?(' '))
      descriptor.strip!
    end
    val = row[2]
    val.strip! if !val.nil? && (val.start_with?(' ') || val.end_with?(' '))
    source_type = row[5]
    if !source_type.nil? && (source_type.start_with?(' ') || source_type.end_with?(' '))
      source_type.strip!
    end
    source_name = row[4]
    if !source_name.nil? && (source_name.start_with?(' ') || source_name.end_with?(' '))
      source_name.strip!
    end
    # create a parameter from this set of data if valid
    if !descriptor.nil? && !val.nil? && !source_type.nil? && !source_name.nil?
      new_param = Parameter.new(descriptor, val, source_type, source_name, csv_var_counter)
      # go ahead and spew it out to the log
      new_param.spew

      # and add it to the array
      parameters.push(new_param)

    end
  end

  # return the entire list
  ParameterList.new(parameters)
end
