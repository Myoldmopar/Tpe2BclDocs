def initialize(map_spreadsheet_object)
    
    # read in the meta data
    owner        = map_spreadsheet_object.cell(3,4) 
    brand        = map_spreadsheet_object.cell(4,4)
    product_line = map_spreadsheet_object.cell(5,4)
    model        = map_spreadsheet_object.cell(6,4)
    
    # instantiate the base class with common data
    super(owner, brand, product_line, model)
    
    # next we need to read in HPWH specific data
    # first instantiate some hashes
    @data_set_master = []
    #@hash_master = Hash.new(nil)
    #[0.25, 0.5, 0.75, 1.0].each { |x| @hash_master[x] = Hash.new(nil) }
    
    # set reading bounds and other parameters, and start reading data
    first_row_of_data = 10
    last_row_of_data  = 37
    column_capacity_fraction = 4
    column_power_in_fraction = 5
    column_heating_cop_frac  = 6
    for row in first_row_of_data..last_row_of_data
        
        # deduce air entering temp algorithmically...user CANT change the template
        case row
        when 10, 17, 24, 31; air_temp = 4.4
        when 11, 18, 25, 32; air_temp = 10
        when 12, 19, 26, 33; air_temp = 15.6
        when 13, 20, 27, 34; air_temp = 21.1
        when 14, 21, 28, 35; air_temp = 26.7
        when 15, 22, 29, 36; air_temp = 32.2
        when 16, 23, 30, 37; air_temp = 37.8
        end
     
        # deduce flow frac algorithmically...user CANT change the template
        case row
        when 10..16; flow_fraction = 0.25
        when 17..23; flow_fraction = 0.5
        when 24..30; flow_fraction = 0.75
        when 31..37; flow_fraction = 1.0
        end
                    
        # read the dependent variables
        capacity_fraction = map_spreadsheet_object.cell(row,column_capacity_fraction)
        power_in_fraction = map_spreadsheet_object.cell(row,column_power_in_fraction)
        heating_cop_frac  = map_spreadsheet_object.cell(row,column_heating_cop_frac)
        
        # TODO: validate inputs, enforcing numerics
        
        # create a new data point
        this_data_point = HPWHMapDataPoint.new(air_temp, 
                                               flow_fraction, 
                                               capacity_fraction, 
                                               power_in_fraction, 
                                               heating_cop_frac)

        # add it to the main dataset hash
        @data_set_master.push(this_data_point)

    end
            
    self.spew("") # print base map data
    HPWHMapDataPoint.spew # print class-level header line
    @data_set_master.each { |x| x.spew } # print all data points data
            
    @num_observations = @data_set_master.length
    
    # create the X matrix, as we'll only need to do that once.
    # this method returns a matrix where each row is [1.0, x_1, x_2], which
    #  imposes a curve fit of the form Y = a + b x_1 + c x_2
    matrix_X = Matrix.rows(get_xy_data)
    
    # each call to CalculateMatrixSystemWithLeastSquares will perform some redundant operations on X
    #  BUT, X is going to be relatively small, and it's likely only doing it three times
    #  I believe it is worth the overhead to keep it all isolated in that little box, and just pass X and y
    
    # calculate least squares coefficients for capacity fraction
    vector_y = Matrix.column_vector(get_z_data_from_string("capacity_fraction"))
    vector_x = Math.calculate_matrix_system_with_least_squares(matrix_X, vector_y)
    Out.write_curve_quadlinear(vector_x, "CapacityFractionCurve")
    
    # calculate least squares coefficients for power input fraction
    vector_y = Matrix.column_vector(get_z_data_from_string("power_in_fraction"))
    vector_x = Math.calculate_matrix_system_with_least_squares(matrix_X, vector_y)
    Out.write_curve_quadlinear(vector_x, "PowerInputFractionCurve")
    
    # calculate least squares coefficients for heating cop fraction
    vector_y = Matrix.column_vector(get_z_data_from_string("heating_cop_frac"))
    vector_x = Math.calculate_matrix_system_with_least_squares(matrix_X, vector_y)
    Out.write_curve_quadlinear(vector_x, "HeatingCOPFractionCurve")
                            
end

def get_xy_data()
    out = []
    @data_set_master.each do |datum|
        out.push([1.0, datum.air_temp, datum.flow_frac])
    end
    return out
end

def get_z_data_from_string(z_string)
    # returns just the single column of data for all observations
    out = []
    @data_set_master.each do |datum|
        out.push(datum.send(z_string.to_sym))
    end
    return out
end
