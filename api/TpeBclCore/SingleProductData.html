def initialize(directory, product_type_string, product_name)
  # keep this for whatever reason
  @process_directory = directory
  @product_name = product_name

  # initialize class csv variables
  @csv_path = ''
  @data_from_csv = nil

  # initialize class map variables
  @map_expected = false
  @maps_found_list = {}
  @data_from_maps = []

  # Search 'products' directory for translator scripts. Skip ProductBaseClass and Rooftop Units for now
  categories = TpeBclCore::Products.module_eval { constants.map { |c| const_get c } }
  categories -= [TpeBclCore::Products::ProductBaseClass]

  # process product
  @product_category = nil
  categories.each do |x|
    @product_category = x if x.product_key == product_type_string
  end

  if @product_category.nil?
    raise ExProductCategoryNotImplemented, "Product category not implemented for key=#{@product_type_string}"
  end

  @skip_this_product = false
  @status_message = 'Successful'
end
