def self.write_idf_object(lighting_watts)

    Out.idf("
  ! Values that must be filled in later:
  !   <zone_name> 
  !   <lights_schedule> 
      
  Lights,
    <zone_name> Lights,      !- Name
    <zone_name>,             !- Zone or ZoneList Name
    <lights_schedule>,       !- Schedule Name
    LightingLevel,           !- Design Level Calculation Method
    #{lighting_watts},       !- Lighting Level {W}
    ,                        !- Watts per Zone Floor Area {W/m2}
    ,                        !- Watts per Person {W/person}
    ,                        !- Return Air Fraction
    ,                        !- Fraction Radiant
    ,                        !- Fraction Visible
    ,                        !- Fraction Replaceable
    ;                        !- End-Use Subcategory
    ")

end
