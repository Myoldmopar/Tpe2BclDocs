def initialize(parent_dir)
    @parent_dir = parent_dir
    unless Dir.exist?(@parent_dir)
        FileUtils.mkdir(@parent_dir)
    end
    @idf_file_path = File.join(@parent_dir, IDF_FILE)
    @idf_file = File.open(@idf_file_path, 'w')
    @idf_file.puts("! TPE -> BCL PRODUCT IDF FILE")
    @idf_written = false
end

def idf_field(field_value, total_spaces=24)
    s_field = field_value.to_s
    pad_length = total_spaces - s_field.length - 1 # -1 for comma
    return s_field + "," + "".ljust(pad_length)
end

def write_one_indep_var_table(xy_data, table_name, unit_types, curve_type = "Cubic")

# make sure there is *something* there first
    if xy_data.length < 1
        $log.log("Attempted to write one indep var table with empty array, returning quietly")
        return
    end

    # check the values first and also get min/max values
    minx, miny = 9999999999, 9999999999
    maxx, maxy = -9999999999, -9999999999
    xy_data.each do |xy|
        if xy.nil?
            return
        elsif xy[0].nil?
            return
        elsif xy[1].nil?
            return
        end
        minx = [minx, xy[0]].min
        miny = [miny, xy[1]].min
        maxx = [maxx, xy[0]].max
        maxy = [maxy, xy[1]].max
    end

    # write the bulk of the idf object
    idf("
Table:OneIndependentVariable,
    #{table_name},   !- Name
    #{curve_type},   !- Curve Type
    LinearInterpolationOfTable, !- Interpolation Type
    #{minx},  !- Minimum Value of x,
    #{maxx}, !- Maximum Value of x,
    #{miny}, !- Minimum Table Output
    #{maxy}, !- Maximum Table Output
    #{unit_types[0]}, !- Input Unit Type for x
    #{unit_types[1]}, !- Output Unit Type
    1.0, !- Normalization Point")

    # write all the values except the last one
    xy_data[0..-2].each do |xy|
        idf("    #{xy[0]}, #{xy[1]},")
    end

    # write the final value terminating it with a semicolon
    idf("    #{xy_data[-1][0]}, #{xy_data[-1][1]};")

end
