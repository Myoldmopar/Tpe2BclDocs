def self.get_source_type_from_dir_name(dir_name)
    if dir_name.end_with?('manu')
        return SOURCE_MANUFACTURER
    elsif dir_name.end_with?('eval')
        return SOURCE_EVALUATOR
    elsif dir_name.end_with?('lab')
        return SOURCE_THIRDPARTYLAB
    else
        $log.log("A data source directory name was encountered that was mal-named", true)
        $log.log("Directory names should end with: manu, eval, or lab", true)
        $log.log("Directory name encountered = " + dir_name.to_s, true)
        $log.log("Aborting.", true)
        raise ExBadDataSourceDirName, "A data source directory name was encountered that was mal-named"
    end
end
