def initialize(parent_dir)
  @parent_dir = parent_dir
  FileUtils.mkdir(@parent_dir) unless Dir.exist?(@parent_dir)
  @log_file_path = File.join(@parent_dir, LOG_FILE)
  @log_file = File.open(@log_file_path, 'w')
  log('TPE -> BCL PRODUCT LOG FILE')
end

def issue_missing_parameter_error(list_of_missing_parameters)
  log('This product is missing key performance parameter definitions which make it unsuitable for pushing to the BCL.')
  log('Missing parameters include:')
  list_of_missing_parameters.each do |param|
    log("  #{param}")
  end
  string = "This product is missing key performance parameter definitions which make it unsuitable for pushing to the BCL.\n"
  string += "Missing parameters include:\n"
  list_of_missing_parameters.each do |param|
    string += "  #{param}\n"
  end
  string
end

def log(message, also_to_console = false)
  @log_file.puts(prefix + message)
  puts message if also_to_console
end
